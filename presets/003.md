# Neon Swarm - Geometric Fast Flock

```javascript
// Neon Swarm - Geometric Fast Flock

// ---- audio ----
a.setBins(8);
a.setSmooth(0.82);
const low  = () => a.fft[0];
const mid  = () => a.fft[3];
const high = () => a.fft[5];
const amp  = () => (low()+mid())*0.6;

// ---- flock params ----
const DEPTH = 4;
const COLS  = 8;
const ROT_S = 0.12;
const FLOW  = 0.2;

// angular geometric birds (triangles/diamonds) — not circles like 004
const bird = (i) => shape(3 + i*2, 0.03, 0.01);

const flowX = (i,d) => Math.sin(time*0.2 + i*1.1) * FLOW*(1-d) * (1 + low()*0.5);
const flowY = (i,d) => Math.cos(time*0.17 + i*0.9) * (FLOW*0.8)*(1-d) * (1 + mid()*0.5);

// cool palette: cyan, electric blue, green, purple — minimal red
const layerColors = [
  [0.15, 0.9, 1.0],   // cyan
  [0.2, 0.5, 1.0],    // electric blue
  [0.1, 1.0, 0.5],    // green
  [0.5, 0.2, 1.0]     // purple
];

function flockLayer(i){
  const d = i / (DEPTH-1);
  const cols = Math.max(3, COLS - i);
  const c = layerColors[i];
  // tint shifts slowly with time for variety
  const tShift = 0.12 * Math.sin(time*0.08 + i*1.5);
  return bird(i)
    .repeat(cols, 6)
    .scale(() => 0.7 - d*0.1 + Math.sin(time*0.5 + i)*0.03 + amp()*0.04)
    .rotate(() => time*ROT_S * (1 + d*0.3) + Math.sin(time*0.08 + i*0.7)*0.05)
    .scrollX(() => flowX(i,d))
    .scrollY(() => flowY(i,d))
    .modulate(noise(1.0 - d*0.2, 0.03 + d*0.03), 0.12 + d*0.1)
    .color(c[0] + tShift, c[1] + tShift*0.5, c[2])
    .brightness(0.025 - d*0.005);
}

// use all 4 buffers: 2 layers per buffer via .add
flockLayer(0).add(flockLayer(1), 0.6).out(o1);
flockLayer(2).add(flockLayer(3), 0.5).out(o2);

// voronoi texture as a slow-moving backdrop (structural break from 004)
const backdrop = () => voronoi(()=> 4 + 2*Math.sin(time*0.05), 0.02, 0.5)
  .color(0.05, 0.12, 0.2)
  .modulate(noise(0.3, 0.01), 0.06)
  .brightness(0.01);

const baseComp = () => backdrop()
  .add(src(o1), 0.9)
  .add(src(o2), 0.7)
  .contrast(1.03)
  .saturate(1.15);

baseComp().out(o0);

// cool-toned aberration: cyan R-shift, blue-green G, purple B
const aberr = () => src(o0)
  .r()
    .scrollX(0.0015 + low()*0.001)
    .scrollY(0.001 + mid()*0.0008)
    .color(0.3, 0.7, 1.0)
  .add(
    src(o0).g()
      .scrollX(-(0.0012 + low()*0.0008))
      .scrollY(-(0.001 + mid()*0.0006))
      .color(0.1, 1.0, 0.5)
  )
  .add(
    src(o0).b()
      .scrollY(0.0014 + amp()*0.001)
      .color(0.4, 0.15, 1.0)
  )
  .saturate(1.2)
  .colorama(0.003 + amp()*0.005);

// feedback trail with modulateHue for shifting colors over time
const trail = () => src(o0)
  .blend(baseComp(), 0.5 + amp()*0.2)
  .modulateScale(noise(1.3, 0.03), 0.004 + amp()*0.001)
  .modulateRotate(noise(0.4, 0.02), 0.003)
  .hue(()=> 0.003*Math.sin(time*0.06))
  .luma(0.005, 0.08)
  .brightness(0.025);

aberr()
  .blend(trail(), 0.7)
  .contrast(1.04)
  .saturate(1.5)
  .out();
```
