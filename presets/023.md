AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.4, 0.03);
var mid = env(()=> a.fft[1]||0, 0.35, 0.04);

// crystal — refraction + specular + prismatic color
setFunction({
  name: 'crystal',
  type: 'src',
  inputs: [
    {name: 'scale', type: 'float', default: 2.5},
    {name: 'spd', type: 'float', default: 0.04},
    {name: 'hueShift', type: 'float', default: 0.0}
  ],
  glsl: `
    vec2 uv = _st * scale;
    float t = time * spd;

    // surface warp — simulates curved glass
    vec2 q = vec2(
      sin(uv.x*1.4 + uv.y*1.8 + t) + sin(uv.y*1.6 + t*0.6),
      cos(uv.y*1.5 + uv.x*1.2 + t*0.7) + sin(uv.x*1.9 + t*0.4)
    ) * 0.5;
    vec2 r = vec2(
      cos((uv.x+q.x)*1.6 + (uv.y+q.y)*1.2 + t*0.3),
      sin((uv.y+q.y)*1.3 + (uv.x+q.x)*1.8 + t*0.2)
    ) * 0.5;

    // fake surface normal from warp gradient
    vec2 norm = normalize(q + r * 0.7 + vec2(0.001));

    // prismatic refraction — each RGB channel bends differently
    float spread = 0.12;
    vec2 uvR = _st + norm * (spread * 1.0);
    vec2 uvG = _st + norm * (spread * 0.6);
    vec2 uvB = _st + norm * (spread * 0.2);

    // rich background sampled at refracted positions
    float bgR = sin(uvR.x*8.0 + uvR.y*5.0 + t*0.3 + q.x*4.0)*0.5 + 0.5;
    float bgG = sin(uvG.x*7.0 + uvG.y*6.0 + t*0.25 + q.y*3.5)*0.5 + 0.5;
    float bgB = sin(uvB.x*6.0 + uvB.y*7.0 + t*0.2 + r.x*4.0)*0.5 + 0.5;

    // second octave for richness
    bgR += sin(uvR.x*3.0 - uvR.y*4.0 + r.y*3.0)*0.3;
    bgG += sin(uvG.x*4.0 + uvG.y*3.0 + r.x*2.5)*0.3;
    bgB += sin(uvB.x*3.5 - uvB.y*3.5 + q.x*3.0)*0.3;

    vec3 refracted = vec3(bgR, bgG, bgB) * 0.4 + 0.15;

    // caustic highlights where light focuses
    vec2 w = uv + q * 2.0;
    float caustic = pow(0.5 + 0.5*sin(w.x*3.14)*sin(w.y*3.14), 6.0);

    // specular — fake light reflection off surface
    float spec = pow(max(dot(norm, normalize(vec2(0.5, 0.7))), 0.0), 12.0);

    // hue rotation
    float hr = hueShift * 6.2831;
    mat3 hm = mat3(
      0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr),
      0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr),
      0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr)
    );

    // ambient floor — no black anywhere
    vec3 ambient = 0.6 + 0.4 * cos(6.2831 * (length(q)*0.3 + hueShift + vec3(0.0, 0.15, 0.4)));

    // combine: strong ambient base + refraction + caustic + specular
    vec3 col = ambient * 0.35 + refracted;
    col += caustic * 0.3;
    col += spec * vec3(0.35, 0.3, 0.4);

    // grain
    float grain = fract(sin(dot(_st*543.31, vec2(12.9898,78.233)))*43758.5453);
    col += (grain - 0.5) * 0.02;

    col = hm * col;
    col = col / (1.0 + col);
    col = pow(col, vec3(0.75));

    return vec4(col, 1.0);
  `
})

crystal(2.5, 0.15, ()=>time*0.003 + mid()*0.003)
