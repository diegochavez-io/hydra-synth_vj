// White Flock - Experimental (setBins 8, amp*1.5)

// ---- audio ----
a.setBins(8);
a.setSmooth(0.9);
const low  = () => a.fft[0];
const mid  = () => a.fft[2];
const amp  = () => (low()+mid())*1.5;

// ---- flock params ----
const DEPTH = 3;
const COLS  = 12;
const EDGE  = 0.015;
const ROUND = 0.02;
const ROT_S = 0.15;
const FLOW  = 0.25;

const bird = () => shape(60, EDGE, ROUND);

const flowX = (i,d) => Math.sin(time*0.3 + i*0.8) * FLOW*(1-d) * (1 + low()*0.4);
const flowY = (i,d) => Math.cos(time*0.25 + i*0.7) * (FLOW*0.6)*(1-d) * (1 + mid()*0.4);

function flockLayer(i){
  const d = i / (DEPTH-1);
  return bird()
    .repeat(
      COLS - i*2,
      () => 14 + Math.sin(time*0.25 + i)*9
    )
    .scale(() => 1 - d*0.15 + Math.sin(time*0.5 + i)*0.02 + amp()*0.02)
    .rotate(() => time*ROT_S * (1 + d*0.15) + Math.sin(time*0.07 + i)*0.02)
    .scrollX(() => flowX(i,d))
    .scrollY(() => flowY(i,d))
    .modulate(noise(0.9 - d*0.3, 0.02 + d*0.05), 0.08 + d*0.12)
    .color( 0.98 - d*0.2, 0.99 - d*0.1, 1.0 )
    .brightness(0.02 - d*0.015);
}

flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

const baseComp = () => src(o1)
  .add(src(o2), 0.85)
  .add(src(o3), 0.6)
  .contrast(1.02)
  .saturate(0.95);

baseComp().out(o0);

// ---- chromatic aberration rainbow tracer ----
const aberr = () => src(o0)
  .r().scrollX( 0.0005 + low()*0.0008 )
      .scrollY( 0.0010 + mid()*0.0006 )
      .color(2.01,0,0)
  .add(
    src(o0).g()
      .scrollX( -(0.0002 + low()*0.0006) )
      .scrollY( -(0.0001 + mid()*0.0005) )
      .color(0,1,0)
  )
  .add(
    src(o0).b()
      .scrollY( 0.0003 + amp()*0.0006 )
      .color(0,0,1)
  )
  .colorama(0.005 + amp()*0.008);

// ---- gentle feedback trail ----
const trail = () => src(o0)
  .blend( baseComp(), 0.5 + amp()*0.1 )
  .modulateScale( noise(1.1, 0.03), 0.003 )
  .modulateRotate( noise(0.8, 0.02), 0.003 )
  .luma(0.04, 0.15)
  .brightness(0.02);

// final: subpixel drift
src(o0).r().scrollX(()=>0.002 + 0.001*low()).scrollY(()=>0.001 + 0.002*mid()).color(1.08, 0.02, 0.01)
  .add( src(o0).g().scrollX(()=>-(0.001 + 0.0006*low())).scrollY(()=>-(0.0012 + 0.0002*mid())).color(0, 1, 0), 0.92 )
  .add( src(o0).b().scrollY(()=>0.0008 + 0.002*amp()).color(0.01, 0.09, 1.0), 0.9 )
  .colorama(()=>0.003 + 0.01*amp())
  .contrast(1.022)
  .saturate(1.2)
  .out();
