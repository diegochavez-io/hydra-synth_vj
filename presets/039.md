# Magnetic Fields

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.9; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);
var high = env(()=> a.fft[3]||0, 0.25, 0.08);

const rampMask = () => Math.min(1, time * 0.8)

// high-frequency osc as raw line texture, heavily warped by noise into flow patterns
osc(80, 0, 0)
  .modulate(noise(2.5, 0.04), 0.15)
  .modulateRotate(noise(1.2, 0.02), 0.08)
  .modulateScale(noise(0.8, 0.01), 0.06)
  .thresh(0.52, 0.15)
  .color(0.9, 0.92, 0.95)
  .brightness(-0.45)
  .out(o1)

// second field layer — different orientation
osc(60, 0, 0.5)
  .rotate(1.57)
  .modulate(noise(3, 0.03), 0.12)
  .modulateRotate(noise(0.9, 0.015), 0.06)
  .thresh(0.5, 0.18)
  .color(0.85, 0.88, 0.92)
  .brightness(-0.5)
  .out(o2)

// composite on dark background with feedback trails
solid(0.02, 0.02, 0.03)
  .add(src(o1), 0.5)
  .add(src(o2), 0.3)
  .blend(
    src(o0)
      .scale(1.001)
      .modulateRotate(noise(0.4, 0.006), 0.002)
      .saturate(0.95)
      .brightness(-0.008),
    () => rampMask() * 0.72
  )
  .color(
    () => 0.85 + high() * 0.15,
    () => 0.88 + mid() * 0.1,
    () => 0.95 + low() * 0.05
  )
  .contrast(1.15)
  .saturate(() => 0.2 + mid() * 0.3)
  .out(o0)

render(o0)
```
