# Oil Film

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);
var high = env(() => a.fft[3] || 0, 0.2, 0.06);

// --- Thin-film interference GLSL
setFunction({
  name: 'thinfilm',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 3.0 },
    { name: 'spd', type: 'float', default: 0.03 },
    { name: 'bandBias', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // undulating surface — generates varying thickness
    vec2 q = vec2(
      sin(uv.x * 1.3 + uv.y * 0.7 + t) + 0.5 * sin(uv.y * 2.1 + t * 0.6),
      cos(uv.y * 1.1 + uv.x * 0.9 + t * 0.8) + 0.5 * cos(uv.x * 1.8 + t * 0.5)
    );
    vec2 r = vec2(
      sin((uv.x + q.x * 0.8) * 1.5 + t * 0.3),
      cos((uv.y + q.y * 0.8) * 1.7 + t * 0.25)
    );

    // film thickness from warp field
    float thickness = (q.x + q.y + r.x * 0.6 + r.y * 0.4) * 0.5 + 2.0;
    thickness += bandBias * 0.5;

    // thin-film interference: path length difference -> phase
    // each wavelength has different phase shift
    float phaseR = thickness * 3.8;
    float phaseG = thickness * 4.5;
    float phaseB = thickness * 5.2;

    // interference intensity: cos^2 of half the phase difference
    float iR = pow(cos(phaseR), 2.0);
    float iG = pow(cos(phaseG), 2.0);
    float iB = pow(cos(phaseB), 2.0);

    // restrained palette: teal, magenta, gold — not full rainbow
    vec3 teal = vec3(0.15, 0.55, 0.55);
    vec3 magenta = vec3(0.55, 0.18, 0.45);
    vec3 gold = vec3(0.65, 0.5, 0.2);
    vec3 dark = vec3(0.08, 0.06, 0.1);

    // mix bands based on interference
    vec3 col = dark;
    col = mix(col, teal, iR * 0.6);
    col = mix(col, magenta, iG * 0.4);
    col = mix(col, gold, iB * 0.35);

    // surface sheen — bright at glancing angles (edge)
    float edgeDist = length(_st - 0.5) * 2.0;
    float sheen = smoothstep(0.3, 1.2, edgeDist) * 0.15;
    col += sheen;

    // subtle variation in surface brightness
    float bright = 0.5 + 0.5 * sin(thickness * 2.0 + t);
    col *= 0.7 + bright * 0.4;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    col = col / (1.0 + col);
    col = pow(col, vec3(0.78));

    return vec4(col, 1.0);
  `
})

const rampMask = () => Math.min(1, time * 0.8)

thinfilm(3.0, 0.03, () => low() * 0.4 - mid() * 0.3 + high() * 0.2)
  .blend(src(o0), () => rampMask() * 0.7)
  .modulateScale(noise(1.0, 0.008), 0.002)
  .saturate(() => 1.1 + 0.04 * mid())
  .colorama(() => 0.001 + 0.001 * high())
  .out(o0)
```
