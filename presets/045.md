# Cloud Chamber

```js
speed = 0.4

AUDIO.bins = 4; AUDIO.smooth = 0.90; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);
var high = env(() => a.fft[3] || 0, 0.2, 0.06);

const rampMask = () => Math.min(1, time * 0.8)

// --- Seed particles (o1): tiny bright dots scattered
shape(99, 0.004, 0.3)
  .repeat(18, 14, () => Math.sin(time * 0.015) * 0.12, () => Math.cos(time * 0.012) * 0.09)
  .modulate(noise(2.5, 0.06), 0.03)
  .color(0.9, 0.92, 1.0)
  .brightness(-0.05)
  .out(o1)

// --- Second particle set (o2): different spacing, offset
shape(99, 0.003, 0.25)
  .repeat(14, 11, () => Math.cos(time * 0.018) * 0.1, () => Math.sin(time * 0.014) * 0.11)
  .modulate(noise(2.0, 0.05), 0.035)
  .scrollX(0.3)
  .scrollY(0.2)
  .color(0.85, 0.88, 1.0)
  .brightness(-0.08)
  .out(o2)

// --- Composite: particles + feedback trails
src(o1)
  .add(src(o2), 0.6)
  .blend(
    src(o0)
      .scale(1.0008)
      .modulateRotate(noise(0.6, 0.015), 0.0015)
      .modulate(noise(1.2, 0.01), 0.001)
      .brightness(-0.003),
    () => rampMask() * 0.88
  )
  .color(
    () => 0.85 + 0.06 * low(),
    () => 0.88 + 0.04 * mid(),
    () => 1.0 + 0.02 * high()
  )
  .contrast(1.12)
  .saturate(() => 0.9 + 0.05 * mid())
  .out(o0)
```
