# Tidal Pool

```js
// Tidal Pool — slow water distortion over warm shallow substrate

AUDIO.bins = 4; AUDIO.smooth = 0.90; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.35, 0.05);
var mid = env(()=> a.fft[1]||0, 0.3, 0.06);

speed = 0.4

// ramp mask for feedback safety
const rampMask = () => Math.min(1, time * 0.8)

// warm sandy substrate — ochre/terracotta/sea green
// noise layers blended to build the sandy bottom
noise(3, 0.02)
  .color(0.85, 0.55, 0.25)
  .add(noise(5, 0.015).color(0.7, 0.35, 0.2), 0.4)
  .add(noise(8, 0.01).color(0.2, 0.5, 0.4), 0.25)
  .saturate(() => 1.1 + low() * 0.08)
  .contrast(1.1)
  .out(o1)

// water ripple modulator — slow noise-driven coordinate warp
noise(2.5, 0.08)
  .modulate(noise(1.5, 0.04), 0.06)
  .out(o2)

// composite: substrate seen through water lens + feedback persistence
src(o1)
  .modulate(src(o2), () => 0.025 + Math.sin(time * 0.07) * 0.008)
  .modulateScale(noise(1.2, 0.03), () => 0.008 + 0.004 * Math.sin(time * 0.05))
  .color(
    () => 0.9 + low() * 0.12,
    () => 0.75 + low() * 0.06,
    () => 0.55 + mid() * 0.04
  )
  .blend(src(o0).scale(1.001).rotate(0.0008), () => rampMask() * 0.72)
  .modulateRotate(noise(0.8, 0.02), 0.003)
  .saturate(1.15)
  .brightness(0.02)
  .out(o0)
```
