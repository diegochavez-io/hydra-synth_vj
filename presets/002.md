```js
// Warm Pulse - Audio-Reactive Amber Flock

// ===== QUICK TWEAKS =====
const BG       = [-0.09, 0.1, 0.03]; // background color [r, g, b]
const TINT     = [1.0, 0.55, 0.2];   // bird tint [r, g, b] — try [0.4, 1.0, 0.8] for teal
const SPEED    = 1.0;                 // global speed multiplier
const DENSITY  = 14;                  // columns of birds (6-24)
const ABERR    = 1.0;                 // chromatic aberration strength (0-3)
const TRAIL_MIX = 0.6;               // trail blend (0=none, 1=heavy)
const SAT      = 1.4;                 // final saturation (0.5-2.0)
const BRIGHT   = 0.02;               // base brightness boost

// ---- audio ----
a.setBins(6);
a.setSmooth(0.8);
const low  = () => a.fft[0];
const mid  = () => a.fft[2];
const high = () => a.fft[4];
const amp  = () => (low()+mid()+high())*0.9;

// ---- flock params ----
const DEPTH = 3;
const EDGE  = 0.042;
const ROUND = 0.28;
const ROT_S = 0.08 * SPEED;
const FLOW  = 0.15 * SPEED;

const bird = () => shape(20, EDGE, ROUND);

const flowX = (i,d) => Math.sin(time*0.15*SPEED + i*0.7) * FLOW*(1-d) * (1 + low()*0.35);
const flowY = (i,d) => Math.cos(time*0.12*SPEED + i*0.8) * (FLOW*0.9)*(1-d) * (1 + mid()*0.3);

function flockLayer(i){
  const d = i / (DEPTH-1);
  return bird()
    .repeat(DENSITY - i*3, () => 10 + Math.sin(time*0.2*SPEED + i)*5)
    .scale(() => 0.9 - d*0.15 + Math.sin(time*0.4*SPEED + i)*0.02 + amp()*0.04)
    .rotate(() => time*ROT_S * (1 + d*0.2) + Math.sin(time*0.09*SPEED + i)*0.03)
    .scrollX(() => flowX(i,d))
    .scrollY(() => flowY(i,d))
    .modulate(noise(1.0 - d*0.25, 0.025 + d*0.04), 0.1 + d*0.1)
    // warm tint fades with depth
    .color(TINT[0], TINT[1] - d*0.25, TINT[2] - d*0.2)
    .brightness(BRIGHT - d*0.01);
}

flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

const baseComp = () => solid(BG[0], BG[1], BG[2])
  .add(src(o1), 1.0)
  .add(src(o2), 1.8)
  .add(src(o3), 1.55)
  .contrast(1.03)
  .saturate(1.15);

baseComp().out(o0);

// warm chromatic aberration
const aberr = () => src(o0)
  .r().scrollX(0.001*ABERR + low()*0.001*ABERR)
    .scrollY(0.0008*ABERR + mid()*0.0006*ABERR)
    .color(1.15, 0.15, 0.05)
  .add(
    src(o0).g()
      .scrollX(-(0.0008*ABERR + low()*0.0005*ABERR))
      .scrollY(-(0.0006*ABERR + mid()*0.0004*ABERR))
      .color(0.3, 1.0, 0.2)
  )
  .add(
    src(o0).b()
      .scrollY(0.0006*ABERR + amp()*0.0008*ABERR)
      .color(0.1, 0.15, 0.7)
  )
  .colorama(0.003 + amp()*0.006);

// audio-reactive hue shift on feedback
const trail = () => src(o0)
  .blend(baseComp(), 0.45 + amp()*0.2)
  .hue(()=> 0.02 * Math.sin(time*0.08*SPEED) + 0.01*low())
  .modulateScale(noise(1.2, 0.04), 0.004 + amp()*0.001)
  .luma(0.03, 0.12)
  .brightness(BRIGHT * 0.8);

aberr()
  .blend(trail(), TRAIL_MIX)
  .contrast(1.04)
  .saturate(SAT)
  .out();
```
