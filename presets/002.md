```js
// Warm Pulse - Evolving Amber Flock

// ===== QUICK TWEAKS =====
const BG        = [-0.09, 0.1, 0.03];
const TINT      = [2.0, -0.55, -0.2];
const SPEED     = 1.0;
const DENSITY   = 0.1;
const ABERR     = 0.9;
const TRAIL_MIX = 1.68;
const SAT       = 1.0;
const BRIGHT    = 0.02;

// ---- audio (heavy smoothing — color warmth only) ----
AUDIO.bins = 6; AUDIO.smooth = 0.92; AUDIO.apply();
var _lo = function(){ return a.fft[0] || 0 };
var _mi = function(){ return a.fft[2] || 0 };
var _hi = function(){ return a.fft[4] || 0 };
var low  = env(_lo, 0.4, 0.04);
var mid  = env(_mi, 0.45, 0.05);
var high = env(_hi, 0.35, 0.06);
var amp  = function(){ return (low() + mid() + high()) * 0.9 };

// ---- flock params ----
const DEPTH = 0.3;
const ROT_S = 0.8 * SPEED;
const FLOW  = 0.05 * SPEED;

// shape morphs slowly: sides drift 6-20, radius and smoothing breathe
const bird = () => shape(
  () => 6 + Math.floor(7 * (1 + Math.sin(time * 0.04 * SPEED))),
  () => 0.32 + 0.12 * Math.sin(time * 0.06 * SPEED + 1.2),
  () => 0.18 + 0.15 * Math.sin(time * 0.05 * SPEED + 2.5)
);

// drift patterns — constant speed, no audio on movement
const flowX = (i, d) =>
  Math.sin(time * 0.15 * SPEED + i * 0.7) * FLOW * (1 - d)
  + Math.sin(time * 0.07 * SPEED + i * 2.3) * 0.04;
const flowY = (i, d) =>
  Math.cos(time * 0.12 * SPEED + i * 0.8) * FLOW * 0.9 * (1 - d)
  + Math.cos(time * 0.055 * SPEED + i * 1.9) * 0.035;

// slowly evolving color per layer — time-driven hue cycling
function layerTint(i, d) {
  var cycle = time * 0.025 * SPEED + i * 0.8;
  var r = TINT[0] + 0.3 * Math.sin(cycle);
  var g = TINT[1] - d * 0.25 + 0.15 * Math.sin(cycle * 0.7 + 1.0);
  var b = TINT[2] - d * 0.2 + 0.2 * Math.sin(cycle * 0.5 + 2.0);
  return [r, g, b];
}

function flockLayer(i) {
  var d = i / (DEPTH - 1);
  var t = layerTint(i, d);
  return bird()
    .repeat(
      () => DENSITY - i * 3 + Math.floor(Math.sin(time * 0.03 * SPEED + i) * 2),
      () => 10 + Math.sin(time * 0.2 * SPEED + i) * 5
    )
    .scale(() => 0.9 - d * 0.15 + Math.sin(time * 0.04 * SPEED + i * 1.4) * 0.04)
    .rotate(() => time * ROT_S * (1 + d * 0.2) + Math.sin(time * 0.035 * SPEED + i) * 0.06)
    .scrollX(() => flowX(i, d))
    .scrollY(() => flowY(i, d))
    .modulate(noise(1.0 - d * 0.25, 0.02 + d * 0.03), 0.1 + d * 0.1)
    .color(() => t[0], () => t[1], () => t[2])
    .brightness(BRIGHT - d * 0.01);
}

flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

const baseComp = () => solid(BG[0], BG[1], BG[2])
  .add(src(o1), 1.0)
  .add(src(o2), 1.8)
  .add(src(o3), 1.55)
  .contrast(1.03)
  .saturate(1.15);

baseComp().out(o0);

// warm chromatic aberration — subtle, no audio on offsets
const aberr = () => src(o0)
  .r().scrollX(0.001 * ABERR).scrollY(0.0008 * ABERR)
    .color(1.15, 0.15, 0.05)
  .add(
    src(o0).g()
      .scrollX(-0.0008 * ABERR).scrollY(-0.0006 * ABERR)
      .color(0.3, 1.0, 0.2)
  )
  .add(
    src(o0).b()
      .scrollY(0.0006 * ABERR)
      .color(0.1, 0.15, 0.7)
  )
  .colorama(() => 0.003 + amp() * 0.004);

// diffusion feedback — reads previous frame, smears with modulateScale + modulateRotate
const trail = () => src(o0)
  .blend(baseComp(), 0.48 + amp() * 0.08)
  .modulateScale(
    noise(1.6, 0.03),
    0.008 + amp() * 0.0008
  )
  .modulateRotate(noise(0.2, 0.02), 0.005)
  .hue(() => 0.03 * Math.sin(time * 0.02 * SPEED) + 0.008 * low())
  .luma(0.005, 0.07)
  .brightness(BRIGHT * 0.8);

aberr()
  .blend(trail(), TRAIL_MIX)
  .contrast(1.04)
  .saturate(() => SAT + amp() * 0.03)
  .out();
```
