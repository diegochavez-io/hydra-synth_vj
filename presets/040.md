# Deep Trench

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.9; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);
var high = env(()=> a.fft[3]||0, 0.25, 0.08);

setFunction({
  name: 'deepTrench',
  type: 'src',
  inputs: [
    {name: 'scale', type: 'float', default: 3.0},
    {name: 'spd', type: 'float', default: 0.04},
    {name: 'bioGlow', type: 'float', default: 0.0}
  ],
  glsl: `
    vec2 uv = _st * scale;
    float t = time * spd;

    // caustic light pattern — overlapping sine waves
    float c1 = sin(uv.x * 3.0 + sin(uv.y * 2.0 + t) * 1.5 + t * 0.3);
    float c2 = sin(uv.y * 2.8 + sin(uv.x * 1.8 + t * 0.7) * 1.3 + t * 0.2);
    float c3 = sin((uv.x + uv.y) * 2.2 + sin(uv.x * 1.5 - t * 0.5) * 1.0 + t * 0.15);
    float caustic = pow(max(0.0, (c1 + c2 + c3) * 0.33 + 0.5), 4.0) * 0.35;

    // deep navy/black base
    vec3 baseColor = vec3(0.01, 0.02, 0.06);

    // caustic teal light on the surface
    vec3 causticColor = vec3(0.05, 0.2, 0.25) * caustic;

    // bioluminescent spots — rare, scattered via hash
    vec2 cellUV = uv * 1.5;
    vec2 cell = floor(cellUV);
    vec2 fUV = fract(cellUV) - 0.5;

    float bioSpot = 0.0;
    vec3 bioColor = vec3(0.0);

    for (int y = -1; y <= 1; y++) {
      for (int x = -1; x <= 1; x++) {
        vec2 neighbor = vec2(float(x), float(y));
        vec2 cCell = cell + neighbor;

        // hash to decide if cell has a spot (sparse)
        float h = fract(sin(dot(cCell, vec2(127.1, 311.7))) * 43758.5453);
        if (h > 0.82) {
          // spot position within cell
          vec2 spotPos = fract(sin(vec2(
            dot(cCell, vec2(269.5, 183.3)),
            dot(cCell, vec2(419.2, 371.9))
          )) * 43758.5453) - 0.5;

          // slow drift
          spotPos += 0.15 * sin(t * 2.0 + h * 6.28);

          float d = length(fUV - neighbor - spotPos);

          // pulsing glow with staggered timing
          float pulse = 0.5 + 0.5 * sin(t * 1.5 + h * 12.0);
          float glow = pulse * smoothstep(0.25, 0.0, d);

          bioSpot += glow;

          // color varies: teal, electric blue, occasional green
          float hue = 0.48 + h * 0.15;
          bioColor += _hsvToRgb(vec3(hue, 0.7, 1.0)) * glow;
        }
      }
    }

    // bioGlow param boosts bioluminescence (audio-driven)
    float bioIntensity = 0.15 + bioGlow * 0.25;

    vec3 col = baseColor + causticColor + bioColor * bioIntensity;

    // very subtle deep gradient — darker at bottom
    col *= 0.8 + 0.2 * _st.y;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.01;

    // tone map — preserve darkness
    col = col / (1.0 + col);
    col = pow(col, vec3(0.85));

    return vec4(col, 1.0);
  `
})

const rampMask = () => Math.min(1, time * 0.8)

deepTrench(3.0, 0.04, () => mid() * 0.4 + high() * 0.3)
  .blend(
    src(o0)
      .scale(1.001)
      .modulate(noise(1.5, 0.01), 0.005)
      .saturate(0.97)
      .brightness(-0.003),
    () => rampMask() * 0.7
  )
  .saturate(() => 1.1 + low() * 0.04)
  .colorama(() => 0.001 + mid() * 0.001)
  .out(o0)

render(o0)
```
