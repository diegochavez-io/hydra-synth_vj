```js
// Daydream Flock — Scope AI feed as background geometry
// 013's chromatic flock + 027's feedback, seeded by Daydream s0
// Requires Daydream connected (s0 = AI feed)

AUDIO.bins = 4; AUDIO.smooth = 0.88; AUDIO.apply();
var _lo = function(){ return a.fft[0] || 0 };
var _mi = function(){ return a.fft[2] || 0 };
var low = env(_lo, 0.25, 0.06);
var mid = env(_mi, 0.3, 0.08);
var amp = function(){ return (low() + mid()) * 0.5 };

// ---- flock params ----
var EDGE = 0.018;
var ROUND = 0.01;
var bird = function(){ return shape(100, EDGE, ROUND) };

// flock layer — birds tinted by depth, flowing gently
function flockLayer(i) {
  var d = i / 2;
  var cols = Math.max(4, 10 - i * 2);
  return bird()
    .repeat(cols, 8)
    .scale(function(){ return 0.5 - d * 0.15 + Math.sin(time * 0.5 + i) * 0.02 + amp() * 0.02 })
    .rotate(function(){ return time * 0.1 * (1 + d * 0.15) + Math.sin(time * 0.07 + i) * 0.02 })
    .scrollX(function(){ return Math.sin(time * 0.12 + i * 0.6) * 0.2 * (1 - d) })
    .scrollY(function(){ return Math.cos(time * 0.10 + i * 0.7) * 0.16 * (1 - d) })
    .modulate(noise(0.9 - d * 0.3, 0.02 + d * 0.05), 0.2 + d * 0.12)
    .brightness(0.018 + d * 0.005)
    .color(
      function(){ return 0.6 + amp() * 0.02 + Math.sin(time * 0.09 + i * 0.8) * 0.3 },
      function(){ return 0.4 + low() * 0.015 + Math.cos(time * 0.07 + i * 1.1) * 0.25 },
      function(){ return 0.7 + mid() * 0.01 + Math.sin(time * 0.11 + i * 0.5) * 0.2 }
    )
}

// flock composite on o2
flockLayer(0).add(flockLayer(1), 0.5).add(flockLayer(2), 0.4)
  .contrast(1.018)
  .saturate(1.05)
  .out(o2)

// chromatic aberration on the flock
src(o2).r().scrollX(0.002).scrollY(0.001).color(1.2, 0.2, 0.3)
  .add(src(o2).g().scrollX(-0.0016).scrollY(-0.001).color(0.3, 1.1, 0.4))
  .add(src(o2).b().scrollY(0.0017).color(0.2, 0.4, 1.3))
  .saturate(1.2)
  .colorama(function(){ return 0.002 + amp() * 0.004 })
  .out(o3)

// Daydream AI feed — luma-keyed as background geometry
src(s0)
  .saturate(1.3)
  .modulate(noise(1.2, 0.02), function(){ return 0.01 + mid() * 0.005 })
  .out(o1)

// final composite: AI background + flock overlay + feedback accumulation
src(o0)
  .blend(src(o1), 0.35)
  .layer(src(o3).mult(solid(1,1,1), 0.25))
  .modulateScale(noise(1.4, 0.02), function(){ return 0.004 + amp() * 0.0004 })
  .modulateRotate(noise(0.3, 0.015), 0.002)
  .hue(function(){ return 0.012 * Math.sin(time * 0.02) + 0.003 * low() })
  .luma(0.008, 0.05)
  .saturate(function(){ return 1.3 + amp() * 0.02 })
  .contrast(1.015)
  .out(o0)

render(o0)
```
