# Ink Diffusion

```js
// Ink Diffusion — fbm domain warping, ink-in-water diffusion
speed = 0.4

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);

const rampMask = () => Math.min(1, time * 0.8)

setFunction({
  name: 'inkDiffuse',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 3.0 },
    { name: 'spd', type: 'float', default: 0.03 },
    { name: 'hueShift', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // fbm-style domain warping — 3 octaves of sin/cos
    vec2 q = vec2(
      sin(uv.x * 1.7 + uv.y * 2.1 + t * 0.8) + cos(uv.y * 1.3 - t * 0.5),
      cos(uv.y * 1.9 - uv.x * 1.4 + t * 0.6) + sin(uv.x * 1.1 + t * 0.3)
    ) * 0.5;

    vec2 r = vec2(
      sin((uv.x + q.x) * 2.3 + (uv.y + q.y) * 1.6 + t * 0.4),
      cos((uv.y + q.y) * 2.1 - (uv.x + q.x) * 1.8 + t * 0.35)
    ) * 0.4;

    vec2 s = vec2(
      cos((uv.x + r.x) * 1.5 + (uv.y + r.y) * 2.4 + t * 0.25),
      sin((uv.y + r.y) * 1.8 - (uv.x + r.x) * 1.3 + t * 0.2)
    ) * 0.3;

    // domain-warped field
    float f = sin((uv.x + s.x) * 1.2 + (uv.y + s.y) * 1.5 + t * 0.15) * 0.5 + 0.5;
    float f2 = sin(length(q + r) * 3.0 + t * 0.3) * 0.5 + 0.5;

    // ink density — darker where warps converge
    float ink = smoothstep(0.15, 0.85, f * 0.6 + f2 * 0.4);

    // muted palette: deep indigo, charcoal, hints of sepia
    vec3 indigo = vec3(0.06, 0.04, 0.14);
    vec3 charcoal = vec3(0.12, 0.11, 0.10);
    vec3 sepia = vec3(0.18, 0.13, 0.08);
    vec3 wash = vec3(0.22, 0.20, 0.24);

    vec3 col = mix(indigo, charcoal, ink);
    col = mix(col, sepia, smoothstep(0.3, 0.7, f2) * 0.4);
    col = mix(col, wash, smoothstep(0.6, 0.95, f) * 0.3);

    // subtle hue rotation for audio
    float hr = hueShift * 6.2831;
    mat3 hm = mat3(
      0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr),
      0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr),
      0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr)
    );
    col = hm * col;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.8));

    return vec4(col, 1.0);
  `
})

// GLSL source with audio on hue only
inkDiffuse(3.5, 0.12, () => time * 0.004 + mid() * 0.005)
  .out(o1)

// Feedback composite
src(o1)
  .blend(src(o0), () => rampMask() * 0.78)
  .modulateScale(noise(1.2, 0.02), 0.003)
  .modulateRotate(noise(0.8, 0.015), 0.001)
  .saturate(() => 1.05 + low() * 0.03)
  .out(o0)
```
