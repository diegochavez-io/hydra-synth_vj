# Liquid Metal

```js
// Liquid Metal — mercury/chrome pool with specular highlights
speed = 0.35

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);

const rampMask = () => Math.min(1, time * 0.8)

setFunction({
  name: 'liquidMetal',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 3.0 },
    { name: 'spd', type: 'float', default: 0.03 },
    { name: 'tempShift', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // undulating surface — layered noise via sin/cos
    vec2 q = vec2(
      sin(uv.x * 2.1 + uv.y * 1.3 + t * 1.5) + cos(uv.y * 1.7 + t * 0.9),
      cos(uv.y * 2.3 - uv.x * 1.5 + t * 1.1) + sin(uv.x * 1.9 + t * 0.7)
    ) * 0.5;

    vec2 r = vec2(
      sin((uv.x + q.x) * 1.8 + (uv.y + q.y) * 2.2 + t * 0.6),
      cos((uv.y + q.y) * 2.0 - (uv.x + q.x) * 1.6 + t * 0.5)
    ) * 0.4;

    // surface height field
    float h = sin(uv.x * 1.5 + q.x * 3.0 + r.x * 2.0 + t * 0.4) * 0.5
            + sin(uv.y * 1.8 + q.y * 2.5 + r.y * 1.5 + t * 0.3) * 0.3
            + sin((uv.x + uv.y) * 2.2 + q.x * 1.5 + t * 0.2) * 0.2;

    // fake normals from height gradient
    float eps = 0.02;
    float hx = sin((uv.x + eps) * 1.5 + q.x * 3.0 + r.x * 2.0 + t * 0.4) * 0.5
             + sin(uv.y * 1.8 + q.y * 2.5 + r.y * 1.5 + t * 0.3) * 0.3
             + sin((uv.x + eps + uv.y) * 2.2 + q.x * 1.5 + t * 0.2) * 0.2;
    float hy = sin(uv.x * 1.5 + q.x * 3.0 + r.x * 2.0 + t * 0.4) * 0.5
             + sin((uv.y + eps) * 1.8 + q.y * 2.5 + r.y * 1.5 + t * 0.3) * 0.3
             + sin((uv.x + uv.y + eps) * 2.2 + q.x * 1.5 + t * 0.2) * 0.2;

    vec2 norm = normalize(vec2(h - hx, h - hy) / eps);

    // specular highlights — sharp chrome reflections
    vec2 lightDir = normalize(vec2(0.4, 0.6));
    float spec = pow(max(dot(norm, lightDir), 0.0), 24.0);
    float spec2 = pow(max(dot(norm, normalize(vec2(-0.3, 0.5))), 0.0), 16.0);

    // broad reflectance
    float refl = (h * 0.5 + 0.5);
    refl = smoothstep(0.2, 0.8, refl);

    // chrome/silver base
    vec3 silver = vec3(0.65, 0.65, 0.68);
    vec3 dark = vec3(0.15, 0.16, 0.20);
    vec3 highlight = vec3(0.95, 0.95, 0.98);

    // blue shadow tint
    vec3 blueShadow = vec3(0.12, 0.14, 0.25);

    vec3 col = mix(dark, silver, refl);
    col = mix(col, blueShadow, (1.0 - refl) * 0.5);
    col += highlight * spec * 0.7;
    col += highlight * spec2 * 0.3;

    // temperature shift from audio — cooler (blue) or warmer (gold)
    vec3 coolShift = vec3(-0.03, -0.01, 0.05);
    vec3 warmShift = vec3(0.05, 0.02, -0.03);
    col += mix(coolShift, warmShift, tempShift);

    // environment reflection — faint curved streaks
    float envRefl = sin(norm.x * 8.0 + norm.y * 6.0 + t * 0.5) * 0.5 + 0.5;
    envRefl = smoothstep(0.4, 0.9, envRefl) * 0.08;
    col += vec3(envRefl);

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.01;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.78));

    return vec4(col, 1.0);
  `
})

// Source — audio drives color temperature
liquidMetal(3.5, 0.12, () => 0.5 + mid() * 0.3)
  .out(o1)

// Feedback — smooth mercury flow
src(o1)
  .blend(src(o0), () => rampMask() * 0.73)
  .modulateScale(noise(1.0, 0.015), 0.003)
  .modulateRotate(noise(0.7, 0.01), 0.0006)
  .saturate(() => 1.0 + low() * 0.02)
  .out(o0)
```
