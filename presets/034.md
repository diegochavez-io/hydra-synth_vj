# Membrane

```js
// Membrane — organic cellular surface with glowing edges
speed = 0.35

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);

const rampMask = () => Math.min(1, time * 0.8)

setFunction({
  name: 'membrane',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 4.0 },
    { name: 'spd', type: 'float', default: 0.03 },
    { name: 'glowInt', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // voronoi-like distance field via point grid
    float minDist = 1.0;
    float secondDist = 1.0;
    vec2 nearestPt = vec2(0.0);

    for (int yi = -1; yi <= 1; yi++) {
      for (int xi = -1; xi <= 1; xi++) {
        vec2 cell = floor(uv) + vec2(float(xi), float(yi));
        // animated cell centers
        vec2 pt = cell + vec2(
          0.5 + 0.4 * sin(cell.x * 1.7 + cell.y * 2.3 + t * 2.0),
          0.5 + 0.4 * cos(cell.y * 1.9 - cell.x * 1.4 + t * 1.7)
        );
        float d = length(uv - pt);
        if (d < minDist) {
          secondDist = minDist;
          minDist = d;
          nearestPt = pt;
        } else if (d < secondDist) {
          secondDist = d;
        }
      }
    }

    // cell walls — edge detection from distance difference
    float edge = secondDist - minDist;
    float wall = 1.0 - smoothstep(0.0, 0.12, edge);

    // soft interior gradient
    float interior = smoothstep(0.0, 0.5, minDist);

    // warm amber/pink palette
    vec3 amber = vec3(0.55, 0.35, 0.15);
    vec3 pink = vec3(0.50, 0.25, 0.30);
    vec3 deep = vec3(0.15, 0.08, 0.06);
    vec3 glowCol = vec3(0.85, 0.55, 0.35);

    // cell interior color
    float cellHash = fract(sin(dot(floor(nearestPt), vec2(127.1, 311.7))) * 43758.5453);
    vec3 col = mix(amber, pink, cellHash * 0.6 + interior * 0.3);
    col = mix(col, deep, interior * 0.5);

    // edge glow — brighter at cell boundaries
    float glow = wall * (0.6 + glowInt * 0.4);
    col += glowCol * glow;

    // subtle pulsing warmth in the membrane
    float breath = sin(t * 3.0 + nearestPt.x * 2.0 + nearestPt.y * 1.5) * 0.5 + 0.5;
    col += vec3(0.04, 0.02, 0.01) * breath;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.78));

    return vec4(col, 1.0);
  `
})

// Source — audio intensifies edge glow color
membrane(4.5, 0.1, () => low() * 0.5)
  .out(o1)

// Feedback — slow organic drift
src(o1)
  .blend(src(o0), () => rampMask() * 0.7)
  .modulateScale(noise(0.9, 0.012), 0.002)
  .modulateRotate(noise(0.6, 0.01), 0.0008)
  .saturate(() => 1.05 + mid() * 0.03)
  .out(o0)
```
