# Aurora

```js
// Aurora — flowing curtains of light, vertical bands drifting
speed = 0.4

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);
var high = env(() => a.fft[3] || 0, 0.2, 0.06);

const rampMask = () => Math.min(1, time * 0.8)

setFunction({
  name: 'aurora',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 2.0 },
    { name: 'spd', type: 'float', default: 0.04 },
    { name: 'hueShift', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // vertically stretched noise bands — curtain shape
    float x = uv.x;
    float y = uv.y;

    // slow horizontal drift via layered sin
    float drift = sin(x * 3.0 + t * 1.2) * 0.3
                + sin(x * 1.7 - t * 0.8) * 0.5
                + sin(x * 5.0 + t * 0.5) * 0.15;

    // vertical shimmer — makes curtains ripple
    float shimmer = sin(y * 8.0 + x * 2.0 + t * 2.0) * 0.08
                  + sin(y * 12.0 - x * 3.0 + t * 1.5) * 0.04;

    // curtain intensity — brighter near top, fading down
    float curtain = smoothstep(0.0, 0.8, 1.0 - y + drift * 0.4 + shimmer);
    curtain *= smoothstep(-0.1, 0.3, y); // fade at very bottom

    // multiple overlapping bands
    float band1 = exp(-pow((x + drift * 0.5 - 0.3) * 2.5, 2.0)) * curtain;
    float band2 = exp(-pow((x - drift * 0.3 - 0.7) * 2.0, 2.0)) * curtain;
    float band3 = exp(-pow((x + sin(t * 0.3) * 0.2 - 0.5) * 3.0, 2.0)) * curtain * 0.6;

    // green/violet/teal palette
    vec3 green = vec3(0.12, 0.55, 0.30);
    vec3 violet = vec3(0.35, 0.12, 0.50);
    vec3 teal = vec3(0.10, 0.45, 0.50);
    vec3 glow = vec3(0.60, 0.85, 0.65); // bright edge glow

    // combine bands with different colors
    vec3 col = vec3(0.02, 0.02, 0.04); // dark sky base
    col += green * band1;
    col += violet * band2;
    col += teal * band3;

    // chromatic edge glow at bright regions
    float edge = pow(curtain, 3.0) * 0.4;
    col += glow * edge * (band1 + band2 * 0.5);

    // hue rotation for audio color shift
    float hr = hueShift * 6.2831;
    mat3 hm = mat3(
      0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr),
      0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr), 0.2-0.2*cos(hr)+0.35*sin(hr),
      0.2-0.2*cos(hr)+0.35*sin(hr), 0.2-0.2*cos(hr)-0.35*sin(hr), 0.6+0.4*cos(hr)
    );
    col = hm * col;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.75));

    return vec4(col, 1.0);
  `
})

// Source — audio shifts curtain colors
aurora(2.5, 0.15, () => time * 0.003 + mid() * 0.004)
  .out(o1)

// Feedback — gentle trailing
src(o1)
  .blend(src(o0), () => rampMask() * 0.7)
  .modulateScale(noise(0.9, 0.015), 0.002)
  .saturate(() => 1.1 + low() * 0.03)
  .colorama(() => 0.001 + high() * 0.001)
  .out(o0)
```
