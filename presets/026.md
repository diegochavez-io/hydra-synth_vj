function playVid(idx) {
  var vid = document.createElement('video');
  vid.crossOrigin = 'anonymous';
  vid.autoplay = true;
  vid.loop = false;
  vid.muted = true;
  vid.src = vids[idx];

    s0.src = vid;
    s0.tex = s0.regl.texture({ data: vid });
    s0.dynamic = true;
    vid.play();
  });
  vid.addEventListener('ended', ()=> {
    vidIdx = (vidIdx + 1) % vids.length;
    playVid(vidIdx);
  });
}
