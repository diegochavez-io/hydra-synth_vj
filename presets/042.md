# Ember Glow

```js
speed = 0.4

// --- Audio
AUDIO.bins = 4; AUDIO.smooth = 0.90; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);
var high = env(() => a.fft[3] || 0, 0.2, 0.06);

const rampMask = () => Math.min(1, time * 0.8)

// --- Ember particles (o1): tiny circles repeated
shape(99, 0.008, 0.4)
  .repeat(25, 18, () => Math.sin(time * 0.02) * 0.1, () => Math.cos(time * 0.015) * 0.08)
  .modulate(noise(3, 0.08), 0.04)
  .scrollY(0, -0.01)
  .color(
    () => 1.8 + 0.3 * low(),
    () => 0.4 + 0.15 * mid(),
    () => 0.04 + 0.03 * high()
  )
  .brightness(-0.1)
  .out(o1)

// --- Secondary ember layer (o2): sparser, bigger, dimmer
shape(99, 0.012, 0.6)
  .repeat(12, 10, () => Math.cos(time * 0.018) * 0.15, () => Math.sin(time * 0.012) * 0.1)
  .modulate(noise(2, 0.05), 0.05)
  .scrollY(0, -0.006)
  .color(
    () => 1.4 + 0.2 * low(),
    () => 0.25 + 0.1 * mid(),
    0.02
  )
  .brightness(-0.15)
  .out(o2)

// --- Composite with feedback trails
src(o1)
  .add(src(o2), 0.5)
  .blend(
    src(o0).scale(1.001).modulate(noise(1.5, 0.02), 0.002),
    () => rampMask() * 0.82
  )
  .modulateRotate(noise(0.8, 0.01), 0.001)
  .saturate(() => 1.1 + 0.05 * low())
  .brightness(0.005)
  .contrast(1.15)
  .colorama(() => 0.001 + 0.001 * mid())
  .out(o0)
```
