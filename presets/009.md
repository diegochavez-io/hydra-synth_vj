# Vortex - Slow Diffusion

```javascript
// Vortex - Slow Diffusion

// ===== COLOR CONTROL PANEL =====
const COLOR_PALETTE = {
  primary: { h: 0.6, s: 0.8, l: 0.7 },
  secondary: { h: 0.1, s: 0.9, l: 0.6 },
  accent: { h: 0.8, s: 0.9, l: 0.8 },
  highlight: { h: 0.9, s: 0.8, l: 0.6 }
};

const COLOR_CONTROLS = {
  hueShift: 1.5,
  saturation: 1.5,
  brightness: 0.8,
  contrast: 1.025,
  audioResponse: 0.9,
  colorSpeed: 1.9,
  depthFade: 0.2,
  rainbowMode: true
};

const DIFFUSION_CONTROLS = {
  mainScale: 0.9,
  mainSpeed: 0.02,
  mainAmount: 0.15,
  feedbackAmount: 0.688,
  feedbackScale: 0.003,
  feedbackRotate: 0.005,
  audioFeedback: 0.15,
  audioScale: 0.0008,
  lumaAmount: 0.005,
  lumaRange: 0.07
};

const ABERRATION_COLORS = {
  red: [1.2, 0.2, 0.3],
  green: [0.3, 1.1, 0.4],
  blue: [0.2, 0.4, 1.3]
};

// ===== HELPER FUNCTIONS =====
function hslToRgb(h, s, l) {
  const c = (1 - Math.abs(2 * l - 1)) * s;
  const x = c * (1 - Math.abs(((h * 6) % 2) - 1));
  const m = l - c / 2;

  let r, g, b;
  if (h < 1/6) { r = c; g = x; b = 0; }
  else if (h < 2/6) { r = x; g = c; b = 0; }
  else if (h < 3/6) { r = 0; g = c; b = x; }
  else if (h < 4/6) { r = 0; g = x; b = c; }
  else if (h < 5/6) { r = x; g = 0; b = c; }
  else { r = c; g = 0; b = x; }

  return [(r + m), (g + m), (b + m)];
}

function getPaletteColor(colorName, timeShift = 0, depthFactor = 0) {
  const baseColor = COLOR_PALETTE[colorName];
  if (!baseColor) return [1, 1, 1];

  let h = baseColor.h;

  if (COLOR_CONTROLS.rainbowMode) {
    h = (time * COLOR_CONTROLS.colorSpeed * 0.1 + timeShift) % 1;
  } else {
    h = (h + time * COLOR_CONTROLS.hueShift * COLOR_CONTROLS.colorSpeed + timeShift) % 1;
  }

  const s = Math.max(0, Math.min(1, baseColor.s * COLOR_CONTROLS.saturation * (1 - depthFactor * COLOR_CONTROLS.depthFade * 0.3)));
  const l = Math.max(0, Math.min(1, baseColor.l * COLOR_CONTROLS.brightness * (1 - depthFactor * COLOR_CONTROLS.depthFade * 0.2)));

  return hslToRgb(h, s, l);
}

// ===== ORIGINAL CODE WITH COLOR MODIFICATIONS =====
AUDIO.bins = 4; AUDIO.smooth = 0.85; AUDIO.apply();
var _lo = function(){ return a.fft[0] || 0 };
var _mi = function(){ return a.fft[2] || 0 };
var low = env(_lo, 0.25, 0.06);
var mid = env(_mi, 0.3, 0.08);
var amp = function(){ return (low() + mid()) * 0.5 };

// ---- flock params ----
const DEPTH = 3;
const COLS = 10;
const EDGE = 0.009;
const ROUND = 0.02;
const ROT_S = 0.1;
const FLOW = 0.1;
const bird = () => shape(200, EDGE, ROUND);

function birdColor(i, d) {
  const colorNames = ['primary', 'secondary', 'accent', 'highlight'];
  const colorIndex = i % colorNames.length;
  const selectedColor = colorNames[colorIndex];

  const baseColor = getPaletteColor(selectedColor, i * 0.2, d);

  const audioMod = COLOR_CONTROLS.audioResponse;
  const r = baseColor[0] + amp() * audioMod * 0.001;
  const g = baseColor[1] + low() * audioMod * 0.001;
  const b = baseColor[2] + mid() * audioMod * 0.001;

  return [
    Math.max(0, Math.min(1, r)),
    Math.max(0, Math.min(1, g)),
    Math.max(0, Math.min(1, b))
  ];
}

// flow fields (tiny, slow)
const flowX = (i, d) => Math.sin(time * 0.12 + i * 0.6) * FLOW * (1 - d) * (1 + low() * 0.2);
const flowY = (i, d) => Math.cos(time * 0.10 + i * 0.7) * (FLOW * 0.8) * (1 - d) * (1 + mid() * 0.2);

function flockLayer(i) {
  const d = i / (DEPTH - 1);
  const cols = Math.max(4, Math.floor(COLS - i * 2));

  let layer = bird()
    .repeat(cols, 8)
    .scale(() => 0.5 - d * 0.15 + Math.sin(time * 0.5 + i) * 0.02 + amp() * 0.02)
    .rotate(() => time * ROT_S * (1 + d * 0.15) + Math.sin(time * 0.07 + i) * 0.02)
    .scrollX(() => flowX(i, d))
    .scrollY(() => flowY(i, d))
    .modulate(
      noise(DIFFUSION_CONTROLS.mainScale - d * 0.3, DIFFUSION_CONTROLS.mainSpeed + d * 0.05),
      DIFFUSION_CONTROLS.mainAmount + d * 0.12
    )
    .brightness(0.018 + d * 0.005);

  const colTint = () => {
    const c = birdColor(i, d);
    const gain = 0.08 + amp() * COLOR_CONTROLS.audioResponse * 0.01;
    return [
      c[0] + gain * low(),
      c[1] + gain * mid(),
      c[2] + gain * 0.5
    ];
  };

  return layer.color(() => {
    const cc = colTint();
    return cc[0];
  }, () => {
    const cc = colTint();
    return cc[1];
  }, () => {
    const cc = colTint();
    return cc[2];
  });
}

// render layers to internal buffers
flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

// composite
const baseComp = () => src(o1)
  .add(src(o2), 0.25)
  .add(src(o3), 0.3)
  .contrast(COLOR_CONTROLS.contrast)
  .saturate(COLOR_CONTROLS.saturation * 0.7);

baseComp().out(o0);

// ---- enhanced chromatic aberration with custom colors ----
const aberr = () => src(o0)
  .r()
  .scrollX(0.002 + low() * 0.0012)
  .scrollY(0.0012 + mid() * 0.0008)
  .color(ABERRATION_COLORS.red[0], ABERRATION_COLORS.red[1], ABERRATION_COLORS.red[2])
  .add(
    src(o0)
    .g()
    .scrollX(-(0.0016 + low() * 0.0008))
    .scrollY(-(0.0011 + mid() * 0.0007))
    .color(ABERRATION_COLORS.green[0], ABERRATION_COLORS.green[1], ABERRATION_COLORS.green[2])
  )
  .add(
    src(o0)
    .b()
    .scrollY(0.0017 + amp() * 0.0009)
    .color(ABERRATION_COLORS.blue[0], ABERRATION_COLORS.blue[1], ABERRATION_COLORS.blue[2])
  )
  .saturate(COLOR_CONTROLS.saturation * 0.8)
  .colorama(0.002 + amp() * 0.004);

// ---- RESTORED gentle feedback trail (key to diffusion effect) ----
const trail = () => src(o0)
  .blend(baseComp(), 0.51 + amp() * DIFFUSION_CONTROLS.audioFeedback)
  .modulateScale(
    noise(1.6, 0.03),
    DIFFUSION_CONTROLS.feedbackScale + amp() * DIFFUSION_CONTROLS.audioScale
  )
  .modulateRotate(noise(0.2, 0.02), DIFFUSION_CONTROLS.feedbackRotate)
  .luma(DIFFUSION_CONTROLS.lumaAmount, DIFFUSION_CONTROLS.lumaRange)
  .brightness(0.027 * COLOR_CONTROLS.brightness);

// final composite
aberr()
  .blend(trail(), DIFFUSION_CONTROLS.feedbackAmount)
  .contrast(COLOR_CONTROLS.contrast * 1.032)
  .saturate(COLOR_CONTROLS.saturation * 1.59)
  .out();
```
