# Nebula

```js
// Nebula — deep space gas clouds, extremely slow, vast and timeless

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.4, 0.04);
var mid = env(()=> a.fft[1]||0, 0.35, 0.05);

speed = 0.2

const rampMask = () => Math.min(1, time * 0.6)

// gas cloud layer A — deep blue/purple, large scale noise
noise(1.2, 0.008)
  .modulate(noise(0.8, 0.005), 0.04)
  .color(0.12, 0.08, 0.45)
  .saturate(1.3)
  .out(o1)

// gas cloud layer B — crimson/magenta, different noise field
noise(1.5, 0.006)
  .modulate(noise(0.6, 0.004), 0.035)
  .rotate(() => time * 0.003 + 0.5)
  .color(0.4, 0.06, 0.18)
  .saturate(1.2)
  .out(o2)

// stars — tiny bright points scattered densely
shape(99, 0.005, 0.01)
  .repeat(50, 35, 0.01, 0.008)
  .scale(() => 1.0 + Math.sin(time * 0.01) * 0.005)
  .rotate(() => time * 0.001)
  .scrollX(() => Math.sin(time * 0.005) * 0.003)
  .scrollY(() => Math.cos(time * 0.004) * 0.002)
  .color(0.9, 0.85, 0.75)
  .brightness(() => 0.0 + 0.02 * Math.sin(time * 0.08))
  .out(o3)

// composite: nebula clouds blended together + stars on top
src(o1)
  .add(src(o2), 0.6)
  .modulateScale(noise(0.5, 0.003), 0.006)
  .modulate(noise(1.0, 0.005), () => 0.012 + Math.sin(time * 0.008) * 0.004)
  .hue(() => 0.003 * mid())
  .add(src(o3).luma(0.3, 0.05), 0.35)
  .color(
    () => 0.85 + low() * 0.06,
    () => 0.7 + mid() * 0.04,
    () => 1.0
  )
  .saturate(() => 1.15 + low() * 0.04)
  .colorama(() => 0.001 + mid() * 0.0005)
  .blend(src(o0).scale(1.0005).rotate(0.0003), () => rampMask() * 0.78)
  .contrast(1.05)
  .brightness(0.01)
  .out(o0)
```
