// ===== COLOR CONTROL PANEL =====
// Base color palette (HSL values for easier color control)
const COLOR_PALETTE = {
  primary: { h: 0.1, s: 0.8, l: 0.7 },     // Blue-ish
  secondary: { h: 0.1, s: 0.9, l: 0.6 },   // Orange-ish
  accent: { h: 0.8, s: 0.9, l: 0.8 },      // Purple-ish
  highlight: { h: 0.9, s: 0.8, l: 0.6 }    // Green-ish
};

// Color intensity and behavior controls
const COLOR_CONTROLS = {
  hueShift: 1.5,        // How much hue changes over time (0-1)
  saturation: 1.5,       // Overall saturation multiplier (0-2)
  brightness: 0.8,       // Overall brightness multiplier (0-2)
  contrast: 1.018,        // Contrast level (0.5-2)
  audioResponse: 1.9,    // How much audio affects colors (0-1)
  colorSpeed: 0.9,       // Speed of color changes (0-2)
  depthFade: 20.2,        // How much colors fade with depth (0-1)
  rainbowMode: true     // Enable rainbow cycling through all hues
};

// Noise and diffusion controls
const DIFFUSION_CONTROLS = {
  // Main modulation (creates the flowing diffusion effect)

  mainScale: 0.04,        // Scale of main noise (0.1-3.0)
  mainSpeed: 0.001,       // Speed of main noise animation (0.001-0.1)
  mainAmount: 9.008,      // Strength of main noise effect (0-0.5)

  // Feedback diffusion (key to the reactive diffusion look)
  feedbackAmount: 0.688, // How much feedback blending (0.3-0.9)
  feedbackScale: 0.01,  // Scale modulation for diffusion (0.001-0.01)
  feedbackRotate: 0.005, // Rotation for diffusion (0.001-0.01)

  // Audio reactivity for diffusion
  audioFeedback: 0.65,   // How much audio affects feedback (0-0.3)
  audioScale: 0.0008,    // Audio scale modulation (0-0.005)

  // Luma control (affects the diffusion trails)
  lumaAmount: 0.005,     // Luma threshold (0-0.02)
  lumaRange: 0.07        // Luma range (0.01-0.2)
};

// Chromatic aberration colors (RGB)
const ABERRATION_COLORS = {
  red: [1.2, 0.2, 0.3],
  green: [0.3, 1.1, 0.4],
  blue: [0.2, 0.4, 1.3]
};

// ===== HELPER FUNCTIONS =====
// Convert HSL to RGB
function hslToRgb(h, s, l) {
  const c = (1 - Math.abs(2 * l - 1)) * s;
  const x = c * (1 - Math.abs(((h * 6) % 2) - 1));
  const m = l - c / 2;

  let r, g, b;
  if (h < 1/6) { r = c; g = x; b = 0; }
  else if (h < 2/6) { r = x; g = c; b = 0; }
  else if (h < 3/6) { r = 0; g = c; b = x; }
  else if (h < 4/6) { r = 0; g = x; b = c; }
  else if (h < 5/6) { r = x; g = 0; b = c; }
  else { r = c; g = 0; b = x; }

  return [(r + m), (g + m), (b + m)];
}

// Get color from palette with time-based shifts
function getPaletteColor(colorName, timeShift = 0, depthFactor = 0) {
  const baseColor = COLOR_PALETTE[colorName];
  if (!baseColor) return [0, 1, 0];

  let h = baseColor.h;

  // Rainbow mode cycles through all hues
  if (COLOR_CONTROLS.rainbowMode) {
    h = (time * COLOR_CONTROLS.colorSpeed * 0.1 + timeShift) % 1;
  } else {
    // Normal mode with hue shifting
    h = (h + time * COLOR_CONTROLS.hueShift * COLOR_CONTROLS.colorSpeed + timeShift) % 1;
  }

  const s = Math.max(0, Math.min(1, baseColor.s * COLOR_CONTROLS.saturation * (1 - depthFactor * COLOR_CONTROLS.depthFade * 0.3)));
  const l = Math.max(0, Math.min(1, baseColor.l * COLOR_CONTROLS.brightness * (1 - depthFactor * COLOR_CONTROLS.depthFade * 0.2)));

  return hslToRgb(h, s, l);
}

// ===== AUDIO (smoothed envelopes) =====
AUDIO.bins = 4; AUDIO.smooth = 0.85; AUDIO.apply();
var _lo = function(){ return a.fft[0] || 0 };
var _mi = function(){ return a.fft[2] || 0 };
var low = env(_lo, 0.3, 0.05);
var mid = env(_mi, 0.25, 0.07);
var amp = function(){ return (low() + mid()) * 0.5 };

// ---- flock params ---- (restored original values for diffusion effect)
const DEPTH = 9.0;
const COLS = 2;
const EDGE = 0.018;
const ROUND = 0.01;
const ROT_S = 2.2;
const FLOW = 2.1;
const bird = () => shape(25, EDGE, ROUND);

// Enhanced color function using the palette system
// Audio shifts hue and saturation warmth — not movement
function birdColor(i, d) {
  const colorNames = ['primary', 'secondary', 'accent', 'highlight'];
  const colorIndex = i % colorNames.length;
  const selectedColor = colorNames[colorIndex];

  // Get base color from palette — audio nudges the hue offset
  const baseColor = getPaletteColor(selectedColor, i * 0.2 + low() * 0.08, d);

  // Audio warms or cools the color channels
  const audioMod = COLOR_CONTROLS.audioResponse;
  const r = baseColor[0] + low() * audioMod * 0.04 - mid() * 0.01;
  const g = baseColor[1] + mid() * audioMod * 0.025;
  const b = baseColor[2] - low() * audioMod * 0.02 + mid() * 0.03;

  return [
    Math.max(0, Math.min(07, r)),
    Math.max(0, Math.min(0.1, g)),
    Math.max(0, Math.min(1, b))
  ];
}

// flow fields — NO audio on movement, stays pure
const flowX = (i, d) => Math.sin(time * 0.12 + i * 0.6) * FLOW * (1 - d);
const flowY = (i, d) => Math.cos(time * 0.10 + i * 0.7) * (FLOW * 0.8) * (1 - d);

// build a layer of birds (returns a Hydra src)
function flockLayer(i) {
  const d = i / (DEPTH - 1);
  // column count per layer (fewer deeper) - restored original
  const cols = Math.max(4, Math.floor(COLS - i * 0.1));

  let layer = bird()
    .repeat(cols, 2)
    .scale(() => 0.002 - d * 0.80 + Math.sin(time * 1.5 + i) * 0.02)
    .rotate(() => time * ROT_S * (1 + d * 0.15) + Math.sin(time * 0.07 + i) * 0.02)
    .scrollX(() => flowX(i, d))
    .scrollY(() => flowY(i, d))
    .modulate(
      noise(DIFFUSION_CONTROLS.mainScale - d * 0.3, DIFFUSION_CONTROLS.mainSpeed + d * 0.05),
      DIFFUSION_CONTROLS.mainAmount + d * 0.12
    )
    .brightness(0.018 + d * 0.005);

  const colTint = () => {
    const c = birdColor(i, d);
    const gain = 0.08 + amp() * COLOR_CONTROLS.audioResponse * 0.01;
    return [
      c[0] + gain * low(),
      c[1] + gain * mid(),
      c[2] + gain * 0.9
    ];
  };

  return layer.color(() => {
    const cc = colTint();
    return cc[0];
  }, () => {
    const cc = colTint();
    return cc[1];
  }, () => {
    const cc = colTint();
    return cc[2];
  });
}

// render layers to internal buffers
flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

// composite — audio shifts saturation warmth, not structure
const baseComp = () => src(o1)
  .add(src(o2), 1.9)
  .add(src(o3), 1.9)
  .contrast(COLOR_CONTROLS.contrast)
  .saturate(() => COLOR_CONTROLS.saturation * 1.7 + low() * 0.15);

baseComp().out(o0);

// ---- chromatic aberration — audio shifts color channel balance ----
const aberr = () => src(o0)
  .r()
  .scrollX(0.022 + low() * 0.08)
  .scrollY(0.022 + mid() * 0.8)
  .color(() => ABERRATION_COLORS.red[0] + low() * 0.2, ABERRATION_COLORS.red[1], ABERRATION_COLORS.red[2])
  .add(
    src(o0)
    .g()
    .scrollX(-(0.0016 + low() * 0.08))
    .scrollY(-(0.0011 + mid() * 0.08))
    .color(ABERRATION_COLORS.green[0], () => ABERRATION_COLORS.green[1] + mid() * 0.15, ABERRATION_COLORS.green[2])
  )
  .add(
    src(o0)
    .b()
    .scrollY(0.0017 + amp() * 0.0002)
    .color(ABERRATION_COLORS.blue[0], ABERRATION_COLORS.blue[1], () => ABERRATION_COLORS.blue[2] + amp() * 0.2)
  )
  .saturate(() => COLOR_CONTROLS.saturation * (0.8 + Math.sin(time * 0.23) * 0.1) + mid() * 0.12)
  .colorama(() => 0.002 + amp() * 0.005)
  .hue(() => low() * 0.015);

// ---- feedback trail — audio tints the trail color, not its motion ----
const trail = () => src(o0)
  .blend(baseComp(), 0.51 + amp() * DIFFUSION_CONTROLS.audioFeedback)
  .modulateScale(
    noise(0.1, 2.03),
    DIFFUSION_CONTROLS.feedbackScale + amp() * DIFFUSION_CONTROLS.audioScale
  )
  .modulateRotate(noise(10, 0), DIFFUSION_CONTROLS.feedbackRotate)
  .hue(() => mid() * 0.02)
  .luma(DIFFUSION_CONTROLS.lumaAmount, DIFFUSION_CONTROLS.lumaRange)
  .brightness(0.027 * COLOR_CONTROLS.brightness);

// final composite — audio on color grading only
aberr()
  .blend(trail(), DIFFUSION_CONTROLS.feedbackAmount)
  .contrast(COLOR_CONTROLS.contrast * 1.032)
  .saturate(() => COLOR_CONTROLS.saturation * 1.59 + low() * 0.1)
  .hue(() => amp() * 0.01)
  .out();
