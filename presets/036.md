# Fog Layers

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.9; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);

const rampMask = () => Math.min(1, time * 0.8)

// layer A — large slow fog bank
noise(3, 0.02)
  .brightness(-0.3)
  .contrast(0.8)
  .color(0.85, 0.85, 0.92)
  .out(o1)

// layer B — finer rolling detail
noise(8, 0.04)
  .brightness(-0.4)
  .contrast(0.7)
  .color(0.78, 0.78, 0.88)
  .modulateScale(noise(1.5, 0.01), 0.08)
  .out(o2)

// layer C — wispy high-frequency layer
noise(18, 0.06)
  .brightness(-0.5)
  .contrast(0.6)
  .color(0.7, 0.7, 0.82)
  .modulate(noise(2, 0.03), 0.02)
  .out(o3)

// composite fog banks + heavy feedback
src(o1)
  .add(src(o2), 0.35)
  .add(src(o3), 0.2)
  .blend(
    src(o0)
      .scale(1.002)
      .scrollX(0.001, 0.01)
      .modulateRotate(noise(0.5, 0.008), 0.003)
      .saturate(0.98),
    () => rampMask() * 0.82
  )
  .color(
    () => 0.72 + low() * 0.12,
    () => 0.72 + low() * 0.06,
    () => 0.82 + mid() * 0.04
  )
  .contrast(1.05)
  .brightness(0.02)
  .saturate(() => 0.4 + mid() * 0.08)
  .out(o0)

render(o0)
```
