```js
// Icy Drift - Cool Blue Flock

// ---- audio ----
a.setBins(4);
a.setSmooth(0.92);
const low  = () => a.fft[0];
const mid  = () => a.fft[2];
const amp  = () => (low()+mid())*0.5;

// ---- flock params ----
const DEPTH = 3;
const COLS  = 26;
const EDGE  = 1.01;
const ROUND = 0.015;
const ROT_S = 0.04;
const FLOW  = 0.08;

const bird = () => shape(150, EDGE, ROUND);

const flowX = (i,d) => Math.sin(time*0.08 + i*0.5) * FLOW*(1-d) * (1 + low()*0.15);
const flowY = (i,d) => Math.cos(time*0.06 + i*0.6) * (FLOW*0.7)*(1-d) * (1 + mid()*0.15);

function flockLayer(i){
  const d = i / (DEPTH-1);
  return bird()
    .repeat(COLS - i*3, () => 12 + Math.sin(time*0.15 + i)*5)
    .scale(() => 1.2 - d*0.12 + Math.sin(time*0.3 + i)*0.015 + amp()*0.01)
    .rotate(() => time*ROT_S * (1 + d*0.1) + Math.sin(time*0.05 + i)*0.01)
    .scrollX(() => flowX(i,d))
    .scrollY(() => flowY(i,d))
    .modulate(noise(0.8 - d*0.2, 0.015 + d*0.03), 0.06 + d*0.08)
    // icy blue-white: front bright cyan, back deeper blue
    .color(-0.8 - d*0.25, -0.92 - d*0.12, 1.0)
    .brightness(0.015 - d*0.01);
}

flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

const baseComp = () => src(o1)
  .add(src(o2), 0.95)
  .add(src(o3), 0.80)
  .contrast(0.3)
  .saturate(1.2);

baseComp().out(o0);

// soft cyan/blue aberration
const aberr = () => src(o0)
  .r().scrollX(0.0004 + low()*0.0005)
    .scrollY(0.0003 + mid()*0.0003)
    .color(0.7, 0.15, 0.1)
  .add(
    src(o0).g()
      .scrollX(-(0.0003 + low()*0.0004))
      .scrollY(-(0.0004 + mid()*0.0002))
      .color(-0.19, -0.9, 0.7), 0.95
  )
  .add(
    src(o0).b()
      .scrollY(0.0004 + amp()*0.0006)
      .color(-0.15, 0.3, 1.1), 0.95
  )
  .colorama(()=>0.001 + 0.004*amp());

// gentle icy trail
const trail = () => src(o0)
  .blend(baseComp(), 0.5 + amp()*0.1)
  .modulateScale(noise(1.2, 0.02), 0.002)
  .luma(0.08, 0.22)
  .brightness(0.02);

aberr()
  .blend(trail(), 0.6)
  .contrast(2.0)
  .saturate(1.1)
  .out();
```
