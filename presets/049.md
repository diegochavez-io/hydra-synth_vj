# Pollen Drift

```js
// Pollen Drift — tiny floating forms in warm spring light

AUDIO.bins = 4; AUDIO.smooth = 0.90; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);

speed = 0.4

const rampMask = () => Math.min(1, time * 0.8)

// near layer — larger pollen, warm gold
shape(99, 0.025, 0.4)
  .repeat(22, 16, () => Math.sin(time * 0.035) * 0.03, () => Math.cos(time * 0.028) * 0.025)
  .scale(() => 1.0 + Math.sin(time * 0.03) * 0.02)
  .rotate(() => time * 0.008)
  .scrollX(() => Math.sin(time * 0.02) * 0.04)
  .scrollY(() => Math.cos(time * 0.017) * 0.03)
  .color(0.75, 0.6, 0.2)
  .out(o1)

// mid layer — smaller, green-gold
shape(99, 0.015, 0.5)
  .repeat(32, 24, () => Math.cos(time * 0.025) * 0.02, () => Math.sin(time * 0.03) * 0.018)
  .scale(() => 0.95 + Math.sin(time * 0.025 + 1.5) * 0.015)
  .rotate(() => time * 0.006 + 0.8)
  .scrollX(() => Math.cos(time * 0.024 + 1.0) * 0.03)
  .scrollY(() => Math.sin(time * 0.02 + 0.5) * 0.025)
  .color(0.55, 0.6, 0.18)
  .out(o2)

// far layer — finest specks, pale gold, most blurred by feedback
shape(99, 0.01, 0.6)
  .repeat(42, 34, () => Math.sin(time * 0.02) * 0.012, () => Math.cos(time * 0.024) * 0.01)
  .scale(() => 0.88 + Math.sin(time * 0.02 + 3.0) * 0.01)
  .rotate(() => time * 0.004 + 2.0)
  .scrollX(() => Math.sin(time * 0.028 + 2.5) * 0.02)
  .scrollY(() => Math.cos(time * 0.022 + 1.8) * 0.018)
  .color(0.65, 0.55, 0.22)
  .out(o3)

// warm light backdrop + all layers + soft feedback glow
solid(0.12, 0.1, 0.04)
  .add(src(o1), 0.55)
  .add(src(o2), 0.4)
  .add(src(o3), 0.3)
  .modulate(noise(1.5, 0.02), 0.006)
  .color(
    () => 1.0 + low() * 0.08,
    () => 0.92 + low() * 0.04,
    () => 0.7 + mid() * 0.03
  )
  .saturate(() => 1.1 + low() * 0.04)
  .colorama(() => 0.001 + mid() * 0.0008)
  .blend(src(o0).scale(1.003).rotate(0.001).saturate(1.02), () => rampMask() * 0.68)
  .brightness(0.015)
  .out(o0)
```
