# Obsidian

```js
// Obsidian — dark volcanic glass with sharp specular highlights

AUDIO.bins = 4; AUDIO.smooth = 0.88; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.25, 0.06);
var mid = env(()=> a.fft[1]||0, 0.3, 0.08);

speed = 0.3

setFunction({
  name: 'obsidian',
  type: 'src',
  inputs: [
    {name: 'sc', type: 'float', default: 3.0},
    {name: 'spd', type: 'float', default: 0.02},
    {name: 'warmth', type: 'float', default: 0.0}
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // layered warp field for glassy surface
    vec2 q = vec2(
      sin(uv.x * 1.7 + uv.y * 2.1 + t) + sin(uv.y * 0.9 + t * 0.7),
      cos(uv.y * 1.8 + uv.x * 1.3 + t * 0.6) + cos(uv.x * 1.1 + t * 0.4)
    ) * 0.5;
    vec2 r = vec2(
      sin((uv.x + q.x) * 2.2 + t * 0.25),
      cos((uv.y + q.y) * 2.0 + t * 0.3)
    ) * 0.4;

    // surface normal from warp gradient
    vec2 norm = normalize(q + r * 2.0 + vec2(0.001));

    // dark glassy base — near black with faint deep purple
    float depth = length(q * 0.6 + r * 0.4);
    vec3 base = vec3(0.02, 0.015, 0.035) + vec3(0.01, 0.005, 0.02) * depth;

    // specular highlights — sharp bright catches
    float spec1 = pow(max(dot(norm, normalize(vec2(0.6, 0.8))), 0.0), 28.0);
    float spec2 = pow(max(dot(norm, normalize(vec2(-0.4, 0.5))), 0.0), 22.0);
    float spec3 = pow(max(dot(norm, normalize(vec2(0.3, -0.7))), 0.0), 35.0);

    // subtle caustic pattern in the glass
    vec2 w = uv + q * 0.8;
    float caustic = pow(0.5 + 0.5 * sin(w.x * 2.5) * sin(w.y * 2.8 + t * 0.2), 8.0);

    // highlight color: mostly white with faint warmth from audio
    vec3 specColor = vec3(0.85, 0.82 + warmth * 0.15, 0.8 + warmth * 0.1);

    vec3 col = base;
    col += spec1 * 0.6 * specColor;
    col += spec2 * 0.35 * specColor;
    col += spec3 * 0.25 * specColor;
    col += caustic * 0.04 * vec3(0.6, 0.55, 0.7);

    // film grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map and gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.8));

    return vec4(col, 1.0);
  `
})

const rampMask = () => Math.min(1, time * 0.8)

obsidian(3.5, 0.06, () => low() * 0.4)
  .blend(src(o0).scale(1.001).rotate(0.0005), () => rampMask() * 0.65)
  .saturate(() => 1.0 + mid() * 0.03)
  .out(o0)
```
