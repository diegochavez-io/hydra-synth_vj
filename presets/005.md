```js
// Flock Pastel Tight - Alternate Colored Variant

// ---- audio (smoothed envelopes) ----
AUDIO.bins = 4; AUDIO.smooth = 0.85; AUDIO.apply();
var _lo = function(){ return a.fft[0] || 0 };
var _mi = function(){ return a.fft[2] || 0 };
var low = env(_lo, 0.25, 0.06);
var mid = env(_mi, 0.3, 0.08);
var amp = function(){ return (low() + mid()) * 0.5 };

// ---- flock params ----
const DEPTH = 3;
const COLS = 10;
const EDGE = 0.010;
const ROUND = 0.02;
const ROT_S = 0.5;
const FLOW = 0.1;

const bird = () => shape(200, EDGE, ROUND);

function birdColor(i, d) {
    const huePhase = i * 0.16 + time * 0.05 + d * 0.6;
    const r = 0.95 + 0.15 * Math.sin(huePhase + 0.0) * (1 - d * 0.7) + amp() * 0.001;
    const g = 0.85 + 0.14 * Math.sin(huePhase + 2.0) * (1 - d * 0.6) + low() * 0.001;
    const b = 0.85 + 0.13 * Math.sin(huePhase + 4.0) * (1 - d * 0.5) + mid() * 0.001;
    return [r, g, b];
}

const flowX = (i, d) => Math.sin(time * 0.12 + i * 0.6) * FLOW * (1 - d) * (1 + low() * 0.2);
const flowY = (i, d) => Math.cos(time * 0.10 + i * 0.7) * (FLOW * 0.8) * (1 - d) * (1 + mid() * 0.2);

function flockLayer(i) {
    const d = i / (DEPTH - 0.1);
    const cols = Math.max(8, Math.floor(COLS - i * 2));
    let layer = bird()
        .repeat(cols, 1)
        .scale(() => 0.5 - d * 0.15 + Math.sin(time * 0.5 + i) * 0.02 + amp() * 0.02)
        .scrollX(() => flowX(i, d))
        .modulate(noise(0.9 - d * 0.3, 0.02 + d * 0.05), 0.08 + d * 0.12)
        .brightness(0.018 + d * 0.005);

    const colTint = () => {
        const idx = i;
        const c = birdColor(idx, d);
        const gain = 0.08 + amp() * 0.01;
        return [c[0] + gain * low(), c[1] + gain * mid(), c[2] + gain * 0.5];
    };

    return layer.color(() => {
        const cc = colTint();
        return cc[0];
    }, () => {
        const cc = colTint();
        return cc[1];
    }, () => {
        const cc = colTint();
        return cc[2];
    });
}

flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

// composite: tighter mix for subtler layering
const baseComp = () => src(o1)
    .add(src(o2), 0.25)
    .add(src(o3), 0.3)
    .contrast(1.01)
    .saturate(1.02);

baseComp().out(o0);

// ---- chromatic aberration rainbow tracer ----
const aberr = () => src(o0)
    .r()
    .scrollX(0.002 + low() * 0.0012)
    .scrollY(0.0012 + mid() * 0.0008)
    .color(1, 0.2, 0.2)
    .add(
        src(o0).g()
        .scrollX(-(0.0016 + low() * 0.0008))
        .scrollY(-(0.0011 + mid() * 0.0007))
        .color(0.4, 1, 0.4)
    )
    .add(
        src(o0).b()
        .scrollY(0.0017 + amp() * 0.0009)
        .color(0.2, 0.3, 1)
    )
    .saturate(1.09)
    .colorama(0.002 + amp() * 0.004);

// ---- gentle feedback trail ----
const trail = () => src(o0)
    .blend(baseComp(), 0.51 + amp() * 0.15)
    .modulateScale(noise(1.6, 0.03), 0.003 + amp() * 0.0008)
    .modulateRotate(noise(0.2, 0.02), 0.002)
    .luma(0.005, 0.07)
    .brightness(0.027);

// final composite
aberr()
    .blend(trail(), 0.688)
    .contrast(1.032)
    .saturate(1.59)
    .out();
```
