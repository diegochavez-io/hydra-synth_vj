# Frost

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.3, 0.04);
var mid = env(() => a.fft[1] || 0, 0.25, 0.05);
var high = env(() => a.fft[3] || 0, 0.2, 0.06);

// --- Crystalline frost: recursive domain-warped noise in GLSL
setFunction({
  name: 'frost',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 3.0 },
    { name: 'spd', type: 'float', default: 0.02 },
    { name: 'hueShift', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // fbm-like layered noise via sine sums
    vec2 p = uv;
    vec2 q = vec2(
      sin(p.x * 1.7 + p.y * 1.3 + t) + sin(p.y * 2.1 + t * 0.7),
      cos(p.y * 1.9 + p.x * 1.1 + t * 0.6) + cos(p.x * 2.3 + t * 0.4)
    ) * 0.5;

    // second warp layer — branching structures
    vec2 r = vec2(
      sin((p.x + q.x * 1.5) * 2.1 + (p.y + q.y) * 1.8 + t * 0.3),
      cos((p.y + q.y * 1.5) * 1.7 + (p.x + q.x) * 2.2 + t * 0.25)
    ) * 0.5;

    // third warp — fine detail
    vec2 s = vec2(
      cos((p.x + r.x * 2.0) * 3.2 + q.y * 1.5 + t * 0.15),
      sin((p.y + r.y * 2.0) * 2.8 + q.x * 1.8 + t * 0.12)
    ) * 0.4;

    // structure intensity from warp field
    float f = length(q) * 0.4 + length(r) * 0.35 + length(s) * 0.25;
    f = smoothstep(0.2, 0.9, f);

    // edge glow — where gradients are steepest
    float dx = length(q + vec2(0.01, 0.0)) - length(q - vec2(0.01, 0.0));
    float dy = length(q + vec2(0.0, 0.01)) - length(q - vec2(0.0, 0.01));
    float edge = length(vec2(dx, dy)) * 8.0;
    edge = smoothstep(0.1, 0.6, edge);

    // cold palette: ice blue, silver, faint violet
    vec3 iceBlue = vec3(0.7, 0.85, 1.0);
    vec3 silver = vec3(0.85, 0.88, 0.92);
    vec3 violet = vec3(0.7, 0.65, 0.9);

    vec3 col = mix(silver * 0.3, iceBlue, f);
    col += edge * vec3(0.3, 0.35, 0.45);
    col = mix(col, violet * 0.5, hueShift * 0.4);

    // faint interior glow
    float glow = pow(f, 3.0) * 0.3;
    col += glow * iceBlue;

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map
    col = col / (1.0 + col);
    col = pow(col, vec3(0.8));

    return vec4(col, 1.0);
  `
})

const rampMask = () => Math.min(1, time * 0.8)

frost(3.5, 0.02, () => 0.15 * high() + 0.1 * mid())
  .blend(src(o0), () => rampMask() * 0.65)
  .modulateScale(noise(1.2, 0.01), 0.002)
  .saturate(() => 1.05 + 0.04 * mid())
  .color(
    () => 0.85 + 0.03 * high(),
    () => 0.9 + 0.02 * mid(),
    () => 1.0 + 0.02 * low()
  )
  .out(o0)
```
