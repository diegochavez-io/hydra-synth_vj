# Paper Grain

```js
speed = 0.2

// --- Audio (barely there)
AUDIO.bins = 4; AUDIO.smooth = 0.95; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.4, 0.03);
var mid = env(() => a.fft[1] || 0, 0.35, 0.04);

// --- Paper fiber texture (o1): multi-scale noise
noise(80, 0.001)
  .mult(noise(30, 0.002), 0.8)
  .modulate(noise(5, 0.003), 0.003)
  .color(0.92, 0.88, 0.82)
  .contrast(1.02)
  .brightness(0.35)
  .out(o1)

// --- Very slow warp field (o2): breathing modulator
noise(2, 0.005)
  .modulate(noise(1.2, 0.002), 0.01)
  .out(o2)

// --- Composite: paper with faintest breathing
src(o1)
  .modulate(src(o2), () => 0.002 + 0.001 * Math.sin(time * 0.03))
  .color(
    () => 0.93 + 0.01 * low(),
    () => 0.89 + 0.008 * low(),
    () => 0.83 + 0.005 * mid()
  )
  .contrast(1.01)
  .brightness(() => 0.02 + 0.003 * low())
  .blend(src(o0), () => Math.min(1, time * 0.5) * 0.6)
  .out(o0)
```
