# Moth Dust

```js
// Moth Dust — soft scattered shapes like moth scales under a microscope

AUDIO.bins = 4; AUDIO.smooth = 0.90; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);

speed = 0.35

const rampMask = () => Math.min(1, time * 0.8)

// layer A — small hexagons drifting, taupe tones
shape(6, 0.08, 0.45)
  .repeat(18, 14, () => Math.sin(time * 0.03) * 0.02, () => Math.cos(time * 0.025) * 0.015)
  .scale(() => 0.9 + Math.sin(time * 0.04) * 0.03)
  .rotate(() => time * 0.02)
  .scrollX(() => Math.sin(time * 0.018) * 0.03)
  .scrollY(() => Math.cos(time * 0.015) * 0.02)
  .color(0.55, 0.48, 0.38)
  .out(o1)

// layer B — smaller, more diffuse, sage green
shape(6, 0.05, 0.6)
  .repeat(24, 20, () => Math.cos(time * 0.02) * 0.015, () => Math.sin(time * 0.028) * 0.02)
  .scale(() => 0.85 + Math.sin(time * 0.035 + 1.5) * 0.025)
  .rotate(() => time * 0.015 + 0.4)
  .scrollX(() => Math.cos(time * 0.022) * 0.025)
  .scrollY(() => Math.sin(time * 0.019) * 0.02)
  .color(0.45, 0.52, 0.35)
  .out(o2)

// layer C — finest, muted gold dust
shape(6, 0.035, 0.7)
  .repeat(30, 26, () => Math.sin(time * 0.017) * 0.01, () => Math.cos(time * 0.023) * 0.012)
  .scale(() => 0.8 + Math.sin(time * 0.03 + 3.0) * 0.02)
  .rotate(() => time * 0.01 + 1.2)
  .scrollX(() => Math.sin(time * 0.025 + 2.0) * 0.02)
  .scrollY(() => Math.cos(time * 0.02 + 1.0) * 0.018)
  .color(0.6, 0.52, 0.28)
  .out(o3)

// composite with feedback glow trails
solid(0.06, 0.05, 0.04)
  .add(src(o1), 0.5)
  .add(src(o2), 0.45)
  .add(src(o3), 0.35)
  .modulate(noise(2, 0.03), 0.008)
  .hue(() => 0.005 * mid())
  .saturate(() => 1.05 + low() * 0.04)
  .colorama(() => 0.001 + mid() * 0.001)
  .blend(src(o0).scale(1.002).rotate(0.001).saturate(1.05), () => rampMask() * 0.7)
  .brightness(0.01)
  .out(o0)
```
