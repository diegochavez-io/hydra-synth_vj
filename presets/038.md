# Stained Glass

```js
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.9; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);

setFunction({
  name: 'stainedGlass',
  type: 'src',
  inputs: [
    {name: 'cellScale', type: 'float', default: 6.0},
    {name: 'spd', type: 'float', default: 0.03},
    {name: 'hueWarmth', type: 'float', default: 0.0}
  ],
  glsl: `
    vec2 uv = _st * cellScale;
    float t = time * spd;

    // animate cell centers with slow drift
    uv += vec2(sin(t * 0.7) * 0.3, cos(t * 0.5) * 0.3);

    // voronoi via distance to nearest random point
    float minDist = 10.0;
    float secondDist = 10.0;
    vec2 nearestCell = vec2(0.0);

    for (int y = -1; y <= 1; y++) {
      for (int x = -1; x <= 1; x++) {
        vec2 cell = floor(uv) + vec2(float(x), float(y));
        // hash for random point inside cell
        vec2 h = fract(sin(vec2(
          dot(cell, vec2(127.1, 311.7)),
          dot(cell, vec2(269.5, 183.3))
        )) * 43758.5453);
        // slow animated jitter
        vec2 pt = cell + 0.5 + 0.35 * sin(t * 0.8 + h * 6.2831);
        float d = length(uv - pt);
        if (d < minDist) {
          secondDist = minDist;
          minDist = d;
          nearestCell = cell;
        } else if (d < secondDist) {
          secondDist = d;
        }
      }
    }

    // edge detection — dark leading between cells
    float edge = smoothstep(0.0, 0.08, secondDist - minDist);

    // unique hue per cell from hash
    float cellHash = fract(sin(dot(nearestCell, vec2(43.23, 71.17))) * 4758.543);

    // warm palette: ambers, reds, deep blues, golds, greens
    // base hue in warm range with spread
    float baseHue = 0.05 + cellHash * 0.35 + hueWarmth * 0.08;
    float sat = 0.55 + cellHash * 0.25;
    float val = 0.65 + 0.2 * sin(cellHash * 12.0 + t);

    vec3 cellColor = _hsvToRgb(vec3(fract(baseHue), sat, val));

    // glow from behind — brighten center of cell
    float glow = 1.0 + 0.3 * (1.0 - minDist * 1.5);

    vec3 col = cellColor * glow * edge;

    // leading is very dark
    col *= edge * 0.85 + 0.15;

    // ambient warmth floor
    col += vec3(0.04, 0.03, 0.02);

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.015;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.8));

    return vec4(col, 1.0);
  `
})

const rampMask = () => Math.min(1, time * 0.8)

stainedGlass(6.0, 0.03, () => low() * 0.15)
  .blend(src(o0).saturate(0.96), () => rampMask() * 0.55)
  .saturate(() => 1.2 + mid() * 0.06)
  .colorama(() => 0.001 + low() * 0.001)
  .out(o0)

render(o0)
```
