# Marble Veins

```js
// Marble Veins — domain-warped sin waves creating stone texture
speed = 0.3

AUDIO.bins = 4; AUDIO.smooth = 0.93; AUDIO.apply();
var low = env(() => a.fft[0] || 0, 0.35, 0.04);
var mid = env(() => a.fft[1] || 0, 0.3, 0.05);

const rampMask = () => Math.min(1, time * 0.8)

setFunction({
  name: 'marble',
  type: 'src',
  inputs: [
    { name: 'sc', type: 'float', default: 3.0 },
    { name: 'spd', type: 'float', default: 0.02 },
    { name: 'warmth', type: 'float', default: 0.0 }
  ],
  glsl: `
    vec2 uv = _st * sc;
    float t = time * spd;

    // iterative domain warping — marble-like veining
    vec2 p = uv;
    for (int i = 0; i < 4; i++) {
      p = vec2(
        sin(p.x * 1.7 + p.y * 1.3 + t + float(i) * 0.7),
        cos(p.y * 1.5 - p.x * 1.1 + t * 0.8 + float(i) * 0.5)
      ) * 0.8 + p * 0.3;
    }

    // vein pattern from warped coordinates
    float vein1 = abs(sin(p.x * 2.5 + p.y * 1.8 + t * 0.3));
    float vein2 = abs(sin(p.y * 3.1 - p.x * 2.2 + t * 0.2));
    float vein3 = abs(sin((p.x + p.y) * 1.9 + t * 0.15));

    // sharp veins through pow
    vein1 = pow(vein1, 3.0);
    vein2 = pow(vein2, 4.0);
    vein3 = pow(vein3, 2.5);

    float veins = vein1 * 0.5 + vein2 * 0.3 + vein3 * 0.2;

    // broad marble flow
    float flow = sin(uv.x * 1.2 + uv.y * 0.8 + p.x * 2.0 + t * 0.1) * 0.5 + 0.5;

    // stone palette — cream to grey to dark veins
    vec3 cream = vec3(0.82, 0.78, 0.72);
    vec3 grey = vec3(0.55, 0.52, 0.50);
    vec3 dark = vec3(0.18, 0.16, 0.15);
    vec3 warm = vec3(0.75, 0.68, 0.58);

    // base stone color
    vec3 col = mix(cream, grey, flow * 0.6);
    // dark veins
    col = mix(col, dark, veins * 0.7);
    // subtle warmth from audio
    col = mix(col, warm, warmth * 0.15);

    // faint secondary veins for detail
    float micro = abs(sin(uv.x * 8.0 + uv.y * 6.0 + p.x * 4.0 + t * 0.05));
    micro = pow(micro, 6.0) * 0.15;
    col -= vec3(micro * 0.7, micro * 0.65, micro * 0.6);

    // grain
    float grain = fract(sin(dot(_st * 543.31, vec2(12.9898, 78.233))) * 43758.5453);
    col += (grain - 0.5) * 0.012;

    // tone map + gamma
    col = col / (1.0 + col);
    col = pow(col, vec3(0.82));

    return vec4(col, 1.0);
  `
})

// Source
marble(3.5, 0.08, () => low() * 0.4)
  .out(o1)

// Feedback — very slow drift
src(o1)
  .blend(src(o0), () => rampMask() * 0.72)
  .modulateScale(noise(0.8, 0.01), 0.002)
  .saturate(() => 1.02 + mid() * 0.02)
  .out(o0)
```
