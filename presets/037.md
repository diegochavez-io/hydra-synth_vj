# Coral Growth

```js
speed = 0.4

AUDIO.bins = 4; AUDIO.smooth = 0.9; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.3, 0.05);
var mid = env(()=> a.fft[1]||0, 0.35, 0.06);

const rampMask = () => Math.min(1, time * 0.6)

// seed shapes — small repeated circles with noise warp for organic branching
shape(99, 0.015, 0.4)
  .repeat(12, 12, 0, 0)
  .modulate(noise(3, 0.08), 0.12)
  .modulateScale(noise(1.5, 0.03), 0.06)
  .scrollX(0, 0.01)
  .scrollY(0, 0.008)
  .color(0.95, 0.45, 0.4)
  .brightness(-0.15)
  .out(o1)

// second seed — offset shapes for branching density
shape(99, 0.01, 0.5)
  .repeat(8, 8, 0.5, 0.5)
  .modulate(noise(4, 0.06), 0.1)
  .modulateRotate(noise(1, 0.02), 0.04)
  .color(0.85, 0.25, 0.3)
  .brightness(-0.2)
  .out(o2)

// composite with heavy feedback — shapes grow outward via scale drift
src(o1)
  .add(src(o2), 0.4)
  .blend(
    src(o0)
      .scale(1.004)
      .modulateRotate(noise(0.6, 0.01), 0.004)
      .modulate(noise(1.8, 0.02), 0.008)
      .saturate(0.97),
    () => rampMask() * 0.83
  )
  .color(
    () => 0.9 + mid() * 0.06,
    () => 0.35 + low() * 0.04,
    () => 0.38 + mid() * 0.03
  )
  .saturate(() => 1.15 + low() * 0.08)
  .contrast(1.08)
  .brightness(0.01)
  .colorama(() => 0.002 + mid() * 0.001)
  .out(o0)

render(o0)
```
