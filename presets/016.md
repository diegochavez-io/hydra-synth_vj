```javascript
// White Flock - Dense Grid, Soft Trail

// ---- audio (very gentle) ----
a.setBins(4);
a.setSmooth(0.9);
const low  = () => a.fft[0];
const mid  = () => a.fft[2];
const amp  = () => (low()+mid())*0.5;

// ---- flock params ----
const DEPTH = 3;
const COLS  = 22;
const EDGE  = 0.010;
const ROUND = 0.012;
const ROT_S = 0.06;
const FLOW  = 0.10;

// base "bird" glyph (diamond-ish)
const bird = () => shape(100, EDGE, ROUND);

// shared, low-frequency flow field (keeps motion unified)
const flowX = (i,d) => Math.sin(time*0.12 + i*0.6) * FLOW*(1-d) * (1 + low()*0.2);
const flowY = (i,d) => Math.cos(time*0.10 + i*0.7) * (FLOW*0.8)*(1-d) * (1 + mid()*0.2);

// per-layer builder (cohesion + alignment, minimal chaos)
function flockLayer(i){
  const d = i / (DEPTH-1);
  return bird()
    .repeat(
      COLS - i*2,
      () => 14 + Math.sin(time*0.25 + i)*9
    )
    .scale(() => 1 - d*0.15 + Math.sin(time*0.5 + i)*0.02 + amp()*0.02)
    .rotate(() => time*ROT_S * (1 + d*0.15) + Math.sin(time*0.07 + i)*0.02)
    .scrollX(() => flowX(i,d))
    .scrollY(() => flowY(i,d))
    .modulate(noise(0.9 - d*0.3, 0.02 + d*0.05), 0.08 + d*0.12)
    .color( 0.98 - d*0.2, 0.99 - d*0.1, 1.0 )
    .brightness(0.021 - d*0.015);
}

// render layers
flockLayer(0).out(o1);
flockLayer(1).out(o2);
flockLayer(2).out(o3);

// composite: crisp, mostly white flock
const baseComp = () => src(o1)
  .add(src(o2), 0.85)
  .add(src(o3), 0.6)
  .contrast(1.02)
  .saturate(0.95);

// draw once
baseComp().out(o0);

// ---- chromatic aberration rainbow tracer ----
const aberr = () => src(o0)
  .r().scrollX( 0.0005 + low()*0.0008 )
      .scrollY( 0.0010 + mid()*0.0006 )
      .color(1.0,0,0)
  .add(
    src(o0).g()
      .scrollX( -(0.0012 + low()*0.0006) )
      .scrollY( -(0.0009 + mid()*0.0005) )
      .color(0,1,0)
  )
  .add(
    src(o0).b()
      .scrollY( 0.0013 + amp()*0.0006 )
      .color(0,0,1)
  )
  .colorama(0.002 + amp()*0.003);

// ---- gentle feedback to leave a short trail, not chaos ----
const trail = () => src(o0)
  .blend( baseComp(), 0.55 + amp()*0.1 )
  .modulateScale( noise(1.6, 0.03), 0.003 )
  .modulateRotate( noise(1.2, 0.02), 0.002 )
  .luma(0.045, 0.2)
  .brightness(0.19);

// final: crisp flock + rainbow edges + short motion trail
aberr()
  .blend( trail(), 0.94 )
  .contrast(1.58)
  .saturate(1.3)
  .out();
```
