AUDIO.bins = 4; AUDIO.smooth = 0.92; AUDIO.apply();
var low = env(()=> a.fft[0]||0, 0.4, 0.03);
var mid = env(()=> a.fft[1]||0, 0.35, 0.04);
var high = env(()=> a.fft[3]||0, 0.3, 0.05);

// caustics — bright flowing light network, dark between
setFunction({
  name: 'caustic',
  type: 'src',
  inputs: [
    {name: 'scale', type: 'float', default: 3.0},
    {name: 'spd', type: 'float', default: 0.03},
    {name: 'hueShift', type: 'float', default: 0.0}
  ],
  glsl: `
    vec2 uv = _st * scale;
    float t = time * spd;

    // domain warp for organic flow
    vec2 q = vec2(
      sin(uv.x*1.7 + uv.y*1.3 + t) + sin(uv.y*2.1 + t*0.7),
      sin(uv.y*1.9 + uv.x*1.1 + t*0.8) + cos(uv.x*2.3 + t*0.5)
    ) * 0.5;
    vec2 r = vec2(
      sin((uv.x+q.x)*1.3 + (uv.y+q.y)*1.7 + t*0.4),
      cos((uv.y+q.y)*1.5 + (uv.x+q.x)*1.1 + t*0.3)
    ) * 0.5;

    // caustic network at three scales
    vec2 w1 = uv + q * 2.0;
    vec2 w2 = uv * 1.3 + r * 1.8;
    vec2 w3 = uv * 0.8 + (q + r) * 1.2;
    float c1 = pow(0.5 + 0.5*sin(w1.x*3.14)*sin(w1.y*3.14), 6.0);
    float c2 = pow(0.5 + 0.5*sin(w2.x*2.7)*sin(w2.y*2.7), 6.0);
    float c3 = pow(0.5 + 0.5*sin(w3.x*3.5)*sin(w3.y*3.5), 6.0);

    // max = no additive mixing, no brown
    float cv = max(c1, max(c2, c3));

    // color from warp geometry — spatial coherence + flow direction
    float warpH = atan(r.y + q.x, r.x + q.y) / 6.2831 + 0.5;
    float warpM = length(q + r);
    float spatial = (uv.x + uv.y) * 0.08 / scale;
    float h = warpH * 0.5 + spatial + warpM * 0.3 + hueShift;

    // cosine palette — always saturated, never brown
    vec3 lineCol = 0.5 + 0.5 * cos(6.2831 * (h + vec3(0.0, 0.15, 0.4)));

    // bright lines = geometry color, dark between
    vec3 col = lineCol * cv * 2.5;

    // subtle depth glow in dark zones
    col += vec3(0.015, 0.008, 0.028) * (1.0 - cv) * warpM;

    col = col / (1.0 + col);
    col = pow(col, vec3(0.78));

    return vec4(col, 1.0);
  `
})
